//| Function: Find FVG Signal                                        |
//| Detects FVGs aligned with a trend filter                         |

TradeSignal Find_FVG_Signal()
{
    TradeSignal sig;
    sig.isValid = false;

    // --- Trend Filter Check ---
    ENUM_POSITION_TYPE trendDirection = GetCurrentTrend();
    if(trendDirection == WRONG_VALUE) return sig; // Don't trade if no clear trend

    MqlRates rates[];
    if(CopyRates(_Symbol, InpFvgTimeframe, 0, 5, rates) < 5) return sig;

    // Shift 0 is the currently forming bar, 1 is the most recent closed bar.
    // An FVG is a 3-bar pattern. We look at bar 1, 2, and 3.
    // The gap exists on the middle candle (bar 2).
    
    // Check for Bullish FVG (after a bullish trend is confirmed)
    if (trendDirection == POSITION_TYPE_BUY && InpBuy) {
        if (rates[3].low > rates[1].high && (rates[3].low - rates[1].high) > InpMinFvgPoints * _Point) {
            
            sig.isValid = true;
            sig.direction = POSITION_TYPE_BUY;
            sig.slPrice = rates[2].low - (InpSLShiftPips * _Point * 10);
            
            // For Limit order, target the high of the first candle
            if(InpOrderType == ORDER_TYPE_LIMIT) {
                sig.orderType = ORDER_TYPE_BUY_LIMIT;
                sig.entryPrice = rates[1].high + (InpShiftValue * _Point * 10);
            } else { // For Market order
                sig.orderType = ORDER_TYPE_BUY;
                sig.entryPrice = SymbolInfoDouble(_Symbol, SYMBOL_ASK);
            }
            
            return sig;
        }
    }
    
    // Check for Bearish FVG (after a bearish trend is confirmed)
    if (trendDirection == POSITION_TYPE_SELL && InpSell) {
        if (rates[1].low > rates[3].high && (rates[1].low - rates[3].high) > InpMinFvgPoints * _Point) {

            sig.isValid = true;
            sig.direction = POSITION_TYPE_SELL;

                sig.orderType = ORDER_TYPE_SELL;
                sig.entryPrice = SymbolInfoDouble(_Symbol, SYMBOL_BID);
            }

            return sig;

//+------------------------------------------------------------------+
//| Execute a Trade                                                  |
//+------------------------------------------------------------------+
void ExecuteTrade(TradeSignal &sig)
  {
   double sl = sig.slPrice;
   double tp = 0;
   
   // --- Calculate Lot Size ---
   double lots = CalculateLotSize(MathAbs(sig.entryPrice - sl));
   if(lots <= 0) 
     {
       Print("Invalid lot size calculated: ", lots, ". Aborting trade.");
       return;
     }

   // --- Calculate Take Profit ---
   double riskDistance = MathAbs(sig.entryPrice - sl);
   if(InpTPType == TP_TYPE_RR)
     {
      tp = (sig.direction == POSITION_TYPE_BUY) ? sig.entryPrice + (riskDistance * InpRRValue) : sig.entryPrice - (riskDistance * InpRRValue);
     }
   else
     {
      tp = (sig.direction == POSITION_TYPE_BUY) ? sig.entryPrice + (InpFixedTPPoints * _Point) : sig.entryPrice - (InpFixedTPPoints * _Point);
     }
   
   // --- Normalize all prices to the correct digits ---
   sl = NormalizeDouble(sl, _Digits);
   tp = NormalizeDouble(tp, _Digits);
   double entry = NormalizeDouble(sig.entryPrice, _Digits);
   
   // --- Place the order based on type ---
   string tradeResult = "";
   switch(sig.orderType)
   {
      case ORDER_TYPE_BUY:
         trade.Buy(lots, _Symbol, entry, sl, tp, InpTradeComment);
         break;
      case ORDER_TYPE_SELL:
         trade.Sell(lots, _Symbol, entry, sl, tp, InpTradeComment);
         break;
      case ORDER_TYPE_BUY_LIMIT:
         trade.BuyLimit(lots, entry, _Symbol, sl, tp, (InpOrderExpiry == 0 ? 0 : TimeCurrent() + InpOrderExpiry * 60));
         break;
      case ORDER_TYPE_SELL_LIMIT:
          trade.SellLimit(lots, entry, _Symbol, sl, tp, (InpOrderExpiry == 0 ? 0 : TimeCurrent() + InpOrderExpiry * 60));
         break;
   }
   
   tradeResult = trade.ResultComment();
   Print(InpTradeComment, ": ", EnumToString(sig.direction), " executed. Result: ", tradeResult);
}

//+------------------------------------------------------------------+
//| Lot Size Calculation                                             |
//+------------------------------------------------------------------+
double CalculateLotSize(double sl_distance_price)
{
    if(InpLotType == LOT_TYPE_FIXED)
    {
        return InpFixedLotSize;
    }
    
    if(InpLotType == LOT_TYPE_RISK_PERCENT)
    {
        if(sl_distance_price <= 0) return 0;
        
        double riskAmount = AccountInfoDouble(ACCOUNT_BALANCE) * (InpRiskPercent / 100.0);
        
        // Calculate the value of 1 lot move by the stop loss distance
        double sl_distance_points = sl_distance_price / _Point;
        double tick_value = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_VALUE);
        double tick_size = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_SIZE);
        double value_per_point = tick_value / tick_size;

        if (value_per_point == 0) return 0.0;

        double money_loss_per_lot = sl_distance_points * value_per_point;
        
        if (money_loss_per_lot == 0) return 0.0;
        
        double lots = NormalizeDouble(riskAmount / money_loss_per_lot, 2);

        // --- Adhere to lot step and min/max lot size for the symbol ---
        double lot_step = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);
        lots = MathFloor(lots / lot_step) * lot_step;

        double min_lot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN);
        double max_lot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MAX);
        
        if(lots < min_lot) lots = min_lot;
        if(lots > max_lot && max_lot > 0) lots = max_lot;

        return lots;
    }
    
    return 0.01; // Default failsafe
}

//| Break Even Management                                            |

void ApplyBreakEven(ulong ticket)
{
    if(!InpBreakEven) return;

    double openPrice = PositionGetDouble(POSITION_PRICE_OPEN);
    double currentPrice = (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY) ? SymbolInfoDouble(_Symbol, SYMBOL_BID) : SymbolInfoDouble(_Symbol, SYMBOL_ASK);
    double slPrice = PositionGetDouble(POSITION_SL);
    double initialRisk = MathAbs(openPrice - slPrice);
    
    double triggerDistance = initialRisk * InpBETriggerRR;
    
    if (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
    {
        if (currentPrice > openPrice + triggerDistance)
        {
            double newSL = openPrice + (InpBufferPoints * _Point * 10);
            if(slPrice < newSL)
            {
                trade.PositionModify(ticket, newSL, PositionGetDouble(POSITION_TP));
            }
        }
    }
    else // Sell
    {
        if (currentPrice < openPrice - triggerDistance)
        {
            double newSL = openPrice - (InpBufferPoints * _Point * 10);
             if(slPrice > newSL || slPrice == 0)
            {
                trade.PositionModify(ticket, newSL, PositionGetDouble(POSITION_TP));
            }
        }
    }
}

    if(InpPTP2_Enabled && PositionGetDouble(POSITION_VOLUME) > SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN))
    {
        CheckPTP(ticket, InpPTP2TriggerRR, InpPTP2RemovePercent / 100.0);
    }
}

void CheckPTP(ulong ticket, double triggerRR, double closePercent)
{
    double openPrice = PositionGetDouble(POSITION_PRICE_OPEN);
    double currentPrice = (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY) ? SymbolInfoDouble(_Symbol, SYMBOL_BID) : SymbolInfoDouble(_Symbol, SYMBOL_ASK);
    double slPrice = PositionGetDouble(POSITION_SL);
    double initialRisk = MathAbs(openPrice - slPrice);
    
    double triggerDistance = initialRisk * triggerRR;
    double volumeToClose = PositionGetDouble(POSITION_VOLUME) * closePercent;
    
    // --- Normalize to lot step ---
    double lot_step = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);
    volumeToClose = MathFloor(volumeToClose / lot_step) * lot_step;
    if(volumeToClose < SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN)) return;


    if (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
    {
        if (currentPrice >= openPrice + triggerDistance)
        {
            trade.PositionClose(ticket, volumeToClose);
        }
    }
    else // Sell
    {
        if (currentPrice <= openPrice - triggerDistance)
        {
            trade.PositionClose(ticket, volumeToClose);
        }
    }
}


//+------------------------------------------------------------------+
//| Trailing Stop Management                                         |
//+------------------------------------------------------------------+
void ApplyTrailingStop(ulong ticket)
{
    if(InpTrailStopType == TSL_TYPE_NONE) return;

    double openPrice = PositionGetDouble(POSITION_PRICE_OPEN);
    double currentPrice = (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY) ? SymbolInfoDouble(_Symbol, SYMBOL_BID) : SymbolInfoDouble(_Symbol, SYMBOL_ASK);
    double slPrice = PositionGetDouble(POSITION_SL);

    if (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
    {
        if(currentPrice - openPrice < InpTStriggerPoints * _Point) return;
        
        double newSL = currentPrice - (InpTSLStepPoints * _Point * 10);
        if(slPrice < newSL)
        {
             trade.PositionModify(ticket, newSL, PositionGetDouble(POSITION_TP));
        }
    }
    else // Sell
    {
        if(openPrice - currentPrice < InpTStriggerPoints * _Point) return;
        
        double newSL = currentPrice + (InpTSLStepPoints * _Point * 10);
        if(slPrice > newSL || slPrice == 0)
        {
             trade.PositionModify(ticket, newSL, PositionGetDouble(POSITION_TP));
        }
    }
}

//+------------------------------------------------------------------+
//| Check and manage daily limits                                  |
//+------------------------------------------------------------------+
bool CheckDailyLimits()
{
   if (!InpDDActive && !InpDPActive) return false;
   
   // Reset at the start of a new day
   if(TimeDay(TimeCurrent()) != TimeDay(dailyLimitStartTime))
   {
      dailyLimitStartTime = TimeCurrent();
   }
   
   double pnl = GetDailyPnl();
   double balance = AccountInfoDouble(ACCOUNT_BALANCE);

   // Check Drawdown Limit
   if(InpDDActive)
   {
      double dd_limit = balance * (InpMaxDailyDD_Percent / 100.0);
      if(pnl < 0 && MathAbs(pnl) >= dd_limit)
      {
         Print("Daily Drawdown Limit of ", InpMaxDailyDD_Percent, "% reached. Closing all trades and stopping for the day.");
         CloseAllTradesByMagic(InpMagic);
         return true; // Stop trading
      }
   }

   // Check Profit Limit
   if(InpDPActive)
   {
      double profit_limit = balance * (InpMaxDailyProfit_Percent / 100.0);
      if(pnl > 0 && pnl >= profit_limit)
      {
         Print("Daily Profit Limit of ", InpMaxDailyProfit_Percent, "% reached. Closing all trades and stopping for the day.");
         CloseAllTradesByMagic(InpMagic);
         return true; // Stop trading
      }
   }
   
   return false;
}

//+------------------------------------------------------------------+
//| Get Profit/Loss for the Current Day                              |
//+------------------------------------------------------------------+
double GetDailyPnl()
{
    ulong ticket = 0;
    double dailyPnl = 0.0;
    datetime from_date = D'1970.01.01'; // Get the server time from struct

    MqlDateTime dt;
    TimeToStruct(TimeCurrent(), dt);
    dt.hour=0; dt.min=0; dt.sec=0;
    from_date = StructToTime(dt);


    HistorySelect(from_date, TimeCurrent());
    
    for(int i = 0; i < HistoryDealsTotal(); i++)
    {
        if((ticket = HistoryDealGetTicket(i)) > 0)
        {
            if(HistoryDealGetInteger(ticket, DEAL_MAGIC) == InpMagic)
            {
               if(HistoryDealGetInteger(ticket, DEAL_ENTRY) == DEAL_ENTRY_IN) {
                    // This is an opening deal, not a closing one
               } else {
                  dailyPnl += HistoryDealGetDouble(ticket, DEAL_PROFIT);
               }
            }
        }
    }

    // Add profit of currently open positions
    for(int i = PositionsTotal() - 1; i >= 0; i--)
    {
        if(PositionSelectByTicket(PositionGetTicket(i)))
        {
             if(PositionGetInteger(POSITION_MAGIC) == InpMagic)
             {
                 dailyPnl += PositionGetDouble(POSITION_PROFIT);
             }
        }
    }

    return dailyPnl;
}

{
    for(int i = PositionsTotal() - 1; i >= 0; i--)
    {
        ulong ticket = PositionGetTicket(i);
        if (ticket > 0 && PositionSelectByTicket(ticket))
        {
             if (PositionGetInteger(POSITION_MAGIC) == magic && PositionGetString(POSITION_SYMBOL) == _Symbol)
            {
                trade.PositionClose(ticket);
            }
        }
    }
    // Optional: Delete pending orders too
    for (int i=OrdersTotal()-1; i>=0; i--) {
        if (OrderSelect(i)) {
             if (OrderGetInteger(ORDER_MAGIC) == magic && OrderGetString(ORDER_SYMBOL) == _Symbol)
             {

             }
        }
    }
}
