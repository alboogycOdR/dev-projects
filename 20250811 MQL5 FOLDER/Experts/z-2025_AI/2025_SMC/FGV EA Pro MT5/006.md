//| MODEL 3: Scan the chart for new High-Quality Order Blocks.       |
//+------------------------------------------------------------------+
void ScanForNewHQOrderBlocks()
{
    int barsToScan = 200; // Limit scan depth for performance
    MqlRates rates[];
    if(CopyRates(_Symbol, _Period, 0, barsToScan, rates) < 5) return;
    ArraySetAsSeries(rates, true);
    
    // We scan from recent history backwards
    for(int i = 3; i < barsToScan; i++)
    {
        // Avoid re-checking bars we've already seen
        if (rates[i].tick_volume > 0 && rates[i].time <= lastBarsAnalyzed) break;

        bool isBearishOB = false;
        bool isBullishOB = false;

        // --- Rule 1: Identify Pivot Candle (last opposing candle that took liquidity) ---
        // Potential Bearish OB: last bullish candle closing ABOVE prior candle's high.
        if (rates[i].close > rates[i].open && rates[i].close > rates[i+1].high)
            isBearishOB = true;

        // Potential Bullish OB: last bearish candle closing BELOW prior candle's low.
        if (rates[i].open > rates[i].close && rates[i].close < rates[i+1].low)
            isBullishOB = true;
            
        // If neither pattern is found, continue to the next bar
        if (!isBearishOB && !isBullishOB) continue;
        
        // --- Now validate the pivot candle with the other rules ---
        bool hasImbalance = false;
        bool hasBOS = false;

        // --- Rule 2: Cause Imbalance (FVG) ---
        if (isBearishOB && rates[i-2].high < rates[i].low) hasImbalance = true;
        if (isBullishOB && rates[i-2].low > rates[i].high) hasImbalance = true;
        
        if(!hasImbalance) continue; // If no FVG, it's not a valid OB.

        // --- Rule 4: Break of Structure (BOS) ---
        if (isBearishOB) {
            double lastSwingLow = iLow(_Symbol, _Period, iLowest(_Symbol, _Period, MODE_LOW, 10, i)); // Simple swing low
            if (rates[i-1].close < lastSwingLow) hasBOS = true;
        }
        if (isBullishOB) {
            double lastSwingHigh = iHigh(_Symbol, _Period, iHighest(_Symbol, _Period, MODE_HIGH, 10, i)); // Simple swing high
            if (rates[i-1].close > lastSwingHigh) hasBOS = true;
        }

        if(!hasBOS) continue; // If it didn't break structure, it's not a valid OB.
        
        // --- If all rules pass, this is a VALID HQ Order Block ---
        // Check if this OB already exists in our database
        bool exists = false;
        for(int j=0; j<ArraySize(m_ob_database); j++) {
            if(m_ob_database[j].time == rates[i].time) {
                exists = true;
                break;
            }
        }
        if(!exists) {
            int newSize = ArraySize(m_ob_database) + 1;
            ArrayResize(m_ob_database, newSize);
            
            m_ob_database[newSize-1].time = rates[i].time;
            m_ob_database[newSize-1].high = rates[i].high;
            m_ob_database[newSize-1].low = rates[i].low;
            m_ob_database[newSize-1].direction = isBullishOB ? 1 : -1;
            m_ob_database[newSize-1].isMitigated = false;
            
            // Draw the valid OB on the chart
            string objName = "FVEAPRO_HQ_OB_" + TimeToString(rates[i].time);
            color objColor = isBullishOB ? bullishOrderBlockColor : bearishOrderBlockColor;
            string objText = isBullishOB ? "Bullish HQ OB" : "Bearish HQ OB";
            CreateRectangle(objName, rates[i].time, rates[i].high, TimeCurrent(), rates[i].low, objColor, objText);

            Print("Found High-Quality OB: ", objText, " at ", TimeToString(rates[i].time));
        }
    }
    
    // Update the last analyzed time to avoid redundant scans
    if (rates[0].tick_volume > 0)
        lastBarsAnalyzed = rates[0].time;
}

//+------------------------------------------------------------------+
//| Find an Entry based on waiting for a return to a HQ Order Block   |
//+------------------------------------------------------------------+
TradeSignal Find_HighQualityOB_Signal()
{
    TradeSignal sig;
    sig.isValid = false;
    MqlRates currentRate[];
    CopyRates(_Symbol, _Period, 0, 1, currentRate);

    // Loop through our database of OBs to find a tradable setup
    for(int i = ArraySize(m_ob_database)-1; i >= 0; i--)
    {
        // Skip if OB is already used or price is too far
        if(m_ob_database[i].isMitigated) continue;

        double entry, sl;
        
        // --- Setup for a BUY trade ---
        if (m_ob_database[i].direction == 1 && InpBuy) {
            entry = m_ob_database[i].high; // Risk entry is placed at the top of a Bullish OB
            // Check if price has come back to mitigate the OB
            if (currentRate[0].close <= entry)
            {
                sig.isValid = true;
                sig.direction = POSITION_TYPE_BUY;
                sig.orderType = (InpOrderType == ORDER_TYPE_MARKET) ? ORDER_TYPE_BUY : ORDER_TYPE_BUY_LIMIT;
                sig.entryPrice = (sig.orderType == ORDER_TYPE_BUY_LIMIT) ? entry : SymbolInfoDouble(_Symbol, SYMBOL_ASK);
                sig.slPrice = m_ob_database[i].low - (InpSLShiftPips * _Point * 10);
                
                // Mark as mitigated and update chart
                m_ob_database[i].isMitigated = true;
                string objName = "FVEAPRO_HQ_OB_" + TimeToString(m_ob_database[i].time);
                ObjectSetInteger(ChartID(), objName, OBJPROP_COLOR, mitigatedOrderBlockColor);
                
                return sig;
            }
        }
        
        // --- Setup for a SELL trade ---
        if (m_ob_database[i].direction == -1 && InpSell) {
            entry = m_ob_database[i].low; // Risk entry is placed at the bottom of a Bearish OB
            // Check if price has come back to mitigate the OB
            if (currentRate[0].close >= entry)
            {
                sig.isValid = true;
                sig.direction = POSITION_TYPE_SELL;
                sig.orderType = (InpOrderType == ORDER_TYPE_MARKET) ? ORDER_TYPE_SELL : ORDER_TYPE_SELL_LIMIT;
                sig.entryPrice = (sig.orderType == ORDER_TYPE_SELL_LIMIT) ? entry : SymbolInfoDouble(_Symbol, SYMBOL_BID);
                sig.slPrice = m_ob_database[i].high + (InpSLShiftPips * _Point * 10);
                
                // Mark as mitigated and update chart
                m_ob_database[i].isMitigated = true;
                string objName = "FVEAPRO_HQ_OB_" + TimeToString(m_ob_database[i].time);
                ObjectSetInteger(ChartID(), objName, OBJPROP_COLOR, mitigatedOrderBlockColor);

                return sig;
            }
        }
    }

    return sig;
}

{
   if(!InpUseStructureFilter) return POSITION_TYPE_BUY; // Can be configured to allow both directions

   double close_val = iClose(_Symbol, InpFilterTimeframe, 1);
   
   if(close_val > sma_val) return POSITION_TYPE_BUY;
   if(close_val < sma_val) return POSITION_TYPE_SELL;
   
   return WRONG_VALUE; // Market is neutral relative to the SMA
}
