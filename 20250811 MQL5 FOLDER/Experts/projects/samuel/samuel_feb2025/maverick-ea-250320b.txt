#define IS_TRANSACTION_ORDER_PLACED            (trans.type == TRADE_TRANSACTION_REQUEST && request.action == TRADE_ACTION_PENDING && OrderSelect(result.order) && (ENUM_ORDER_STATE)OrderGetInteger(ORDER_STATE) == ORDER_STATE_PLACED)
#define IS_TRANSACTION_ORDER_MODIFIED          (trans.type == TRADE_TRANSACTION_REQUEST && request.action == TRADE_ACTION_MODIFY && OrderSelect(result.order) && (ENUM_ORDER_STATE)OrderGetInteger(ORDER_STATE) == ORDER_STATE_PLACED)
#define IS_TRANSACTION_ORDER_DELETED           (trans.type == TRADE_TRANSACTION_HISTORY_ADD && (trans.order_type >= 2 && trans.order_type < 6) && trans.order_state == ORDER_STATE_CANCELED)

#define IS_TRANSACTION_ORDER_TRIGGERED         (trans.type == TRADE_TRANSACTION_HISTORY_ADD && (trans.order_type >= 2 && trans.order_type < 6) && trans.order_state == ORDER_STATE_FILLED)

#define IS_TRANSACTION_POSITION_CLOSED         (trans.type == TRADE_TRANSACTION_DEAL_ADD && HistoryDealSelect(trans.deal) && (ENUM_DEAL_ENTRY)HistoryDealGetInteger(trans.deal, DEAL_ENTRY) == DEAL_ENTRY_OUT && ((ENUM_DEAL_REASON)HistoryDealGetInteger(trans.deal, DEAL_REASON) != DEAL_REASON_SL && (ENUM_DEAL_REASON)HistoryDealGetInteger(trans.deal, DEAL_REASON) != DEAL_REASON_TP))
#define IS_TRANSACTION_POSITION_CLOSEBY        (trans.type == TRADE_TRANSACTION_DEAL_ADD && HistoryDealSelect(trans.deal) && (ENUM_DEAL_ENTRY)HistoryDealGetInteger(trans.deal, DEAL_ENTRY) == DEAL_ENTRY_OUT_BY)
#define IS_TRANSACTION_POSITION_MODIFIED       (trans.type == TRADE_TRANSACTION_REQUEST && request.action == TRADE_ACTION_SLTP)

      //if(HistoryDealGetInteger(deal, DEAL_MAGIC) == EAMAGIC) {
      //   PrintFormat("Position is closed by opposite position. (position %I64u, deal %I64u)", position, deal);
      //}
      if(HistoryDealGetInteger(deal, DEAL_MAGIC) == InpMagicMain)
        {
         if(InpEnableDebug)
            PrintFormat("Position (main) is closed by opposite position. (position %I64u, deal %I64u)", position, deal);
        }
      if(HistoryDealGetInteger(deal, DEAL_MAGIC) == InpMagicRecovery)
        {
         if(InpEnableDebug)
            PrintFormat("Position (recovery) is closed by opposite position. (position %I64u, deal %I64u)", position, deal);
        }
     }
   //+------------------------------------------------------------------+
   //|                                                                  |
   //+------------------------------------------------------------------+
   virtual void      TradeTransactionPositionModified(ulong position)
     {
      if(InpEnableDebug)
         PrintFormat("Position is modified. (position %I64u)", position);
     }
  };

void OnTradeTransaction(const MqlTradeTransaction &trans,
                        const MqlTradeRequest &request,
                        const MqlTradeResult &result)
  {
   //Print("=============================");
   //Print("OnTradeTransaction");
   if(trans.type == TRADE_TRANSACTION_DEAL_ADD)
     {
      //Print("trans.deal=", trans.deal);
      ulong deal_ticket = trans.deal;
      ENUM_DEAL_ENTRY entry = (ENUM_DEAL_ENTRY)HistoryDealGetInteger(deal_ticket, DEAL_ENTRY);
      ENUM_DEAL_REASON reason = (ENUM_DEAL_REASON)HistoryDealGetInteger(deal_ticket, DEAL_REASON);
      //long entry = HistoryDealGetInteger(trans.deal, DEAL_ENTRY);
      long magic = HistoryDealGetInteger(trans.deal, DEAL_MAGIC);
      

      // Print("entry == DEAL_ENTRY_OUT=", (entry == DEAL_ENTRY_OUT));
      // Print("entry=", EnumToString(entry));

      // Print("reason == DEAL_REASON_SL=", (reason == DEAL_REASON_SL));
      // Print("reason=", EnumToString(reason));

      // Print("magic == InpMagicMain=", (magic == InpMagicMain));
      // Print("magic=", magic);

      // Print("InpMagicMain=", InpMagicMain);


      if(entry == DEAL_ENTRY_OUT && reason == DEAL_REASON_SL && magic == InpMagicMain)
        {
         long position_id = HistoryDealGetInteger(deal_ticket, DEAL_POSITION_ID);
         if(HistorySelectByPosition(position_id))
           {
            int total = HistoryDealsTotal();
            for(int i = 0; i < total; i++)
              {
               ulong ticket = HistoryDealGetTicket(i);
               if(HistoryDealGetInteger(ticket, DEAL_ENTRY) == DEAL_ENTRY_IN)
                 {
                  string comment = HistoryDealGetString(ticket, DEAL_COMMENT);
                  //Print("Stop loss triggered, original comment: ", comment);
                  break;
                 }
              }
           }
         //old code
         //Print("entry == DEAL_ENTRY_OUT && magic == InpMagicMain");
         double profit = HistoryDealGetDouble(trans.deal, DEAL_PROFIT);
         string comment = HistoryDealGetString(trans.deal, DEAL_COMMENT);
         ENUM_DEAL_REASON closeReason = (ENUM_DEAL_REASON)HistoryDealGetInteger(trans.deal, DEAL_REASON);
         //guide handling
         string guid = "";
         if(StringSubstr(comment, 0, 1) == "B")
            guid = StringSubstr(comment, 1);
         else
            if(StringSubstr(comment, 0, 1) == "S")
               guid = StringSubstr(comment, 1);
         //Print("guid=", guid);
         // If this is part of a tracked pair
         if(guid != "")
           {
            // if(InpEnableDebug)
            //    Print("Tracked pair deal: ", guid, ", Profit: ", profit,
            //          ", Close reason: ", EnumToString(closeReason),
            //          ", Comment: ", comment);
            // If this is a buy position
            if(StringSubstr(comment, 0, 1) == "B")
              {
               lastClosedPair.guid = guid;
               lastClosedPair.buyProfit = profit;
               lastClosedPair.buyHitSL = (closeReason == DEAL_REASON_SL);
               // if(InpEnableDebug)
               //    Print("BUY side closed with profit: ", profit,
               //          ", SL hit: ", lastClosedPair.buyHitSL ? "true" : "false");
               // Check if we have both sides of the pair closed
               if(lastClosedPair.guid == guid && lastClosedPair.sellProfit != 0)
                 {
                  lastClosedPair.totalProfit = lastClosedPair.buyProfit + lastClosedPair.sellProfit;
                  // if(InpEnableDebug)
                  //    Print("Both sides closed: Trade pair ", guid, " total profit: ",
                  //          lastClosedPair.totalProfit,
                  //          ", buyHitSL: ", lastClosedPair.buyHitSL ? "true" : "false",
                  //          ", sellHitSL: ", lastClosedPair.sellHitSL ? "true" : "false");
                 }
              }
            // If this is a sell position
            else
               if(StringSubstr(comment, 0, 1) == "S")
                 {
                  lastClosedPair.guid = guid;
                  lastClosedPair.sellProfit = profit;
                  lastClosedPair.sellHitSL = (closeReason == DEAL_REASON_SL);
                  // if(InpEnableDebug)
                  //    Print("SELL side closed with profit: ", profit,
                  //          ", SL hit: ", lastClosedPair.sellHitSL ? "true" : "false");
                  // Check if we have both sides of the pair closed
                  if(lastClosedPair.guid == guid && lastClosedPair.buyProfit != 0)
                    {
                     lastClosedPair.totalProfit = lastClosedPair.buyProfit + lastClosedPair.sellProfit;
                     // if(InpEnableDebug)
                     //    Print("Both sides closed: Trade pair ", guid, " total profit: ",
                     //          lastClosedPair.totalProfit,
                     //          ", buyHitSL: ", lastClosedPair.buyHitSL ? "true" : "false",
                     //          ", sellHitSL: ", lastClosedPair.sellHitSL ? "true" : "false");
                    }
                 }
           }
        }
     }
   ExtTransaction.OnTradeTransaction(trans,request,result);
  }

//+------------------------------------------------------------------+
//| Expert initialization function                                    |
//+------------------------------------------------------------------+
int OnInit()
  {
// Initialize the lastFiveTrades array with zero times (indicating no trades)
   for(int i = 0; i < 5; i++)
     {
      lastFiveTrades[i].time = 0;
      lastFiveTrades[i].profit = 0.0;
     }
// Select all trade history from account start to current time
   if(!HistorySelect(0, TimeCurrent()))
     {
      Print("Failed to select trade history");
      return(INIT_FAILED);
     }
// Calculate initial total realized profit from all closed trades with specified magic numbers
   int totalDeals = HistoryDealsTotal();
   for(int i = 0; i < totalDeals; i++)
     {
      ulong dealTicket = HistoryDealGetTicket(i);
      if(HistoryDealSelect(dealTicket))
        {
         long entry = HistoryDealGetInteger(dealTicket, DEAL_ENTRY);
         long magic = HistoryDealGetInteger(dealTicket, DEAL_MAGIC);
         if(entry == DEAL_ENTRY_OUT && (magic == InpMagicRecovery || magic == InpMagicMain))
           {
            totalRealizedProfit += HistoryDealGetDouble(dealTicket, DEAL_PROFIT);
           }
        }
     }
// Fill lastFiveTrades with the most recent 5 closed trades with specified magic numbers
   int count = 0;
   for(int i = totalDeals - 1; i >= 0 && count < 5; i--)
     {
      ulong dealTicket = HistoryDealGetTicket(i);
      if(HistoryDealSelect(dealTicket))
        {
         long entry = HistoryDealGetInteger(dealTicket, DEAL_ENTRY);
         long magic = HistoryDealGetInteger(dealTicket, DEAL_MAGIC);
         if(entry == DEAL_ENTRY_OUT && (magic == InpMagicRecovery || magic == InpMagicMain))
           {
            lastFiveTrades[4 - count].profit = HistoryDealGetDouble(dealTicket, DEAL_PROFIT);
            lastFiveTrades[4 - count].time = (datetime)HistoryDealGetInteger(dealTicket, DEAL_TIME);
            count++;
           }
        }
     }
// Set magic number for main trade operations by default
   trade.SetExpertMagicNumber(InpMagicMain);
   trade.LogLevel(LOG_LEVEL_NO);
// Check for valid input parameters
   if(InpLotSize <= 0 || InpSLPointsMain <= 0 || InpTPPointsMain <= 0 ||
      InpSLPointsRecovery <= 0 || InpTPPointsRecovery <= 0)
     {
      Print("Error: Invalid input parameters. Ensure lot size and SL/TP points are positive.");
      return INIT_PARAMETERS_INCORRECT;
     }
// Validate main trade lot size
   double minLot = SymbolInfoDouble(Symbol(), SYMBOL_VOLUME_MIN);
   double maxLot = SymbolInfoDouble(Symbol(), SYMBOL_VOLUME_MAX);
   double stepLot = SymbolInfoDouble(Symbol(), SYMBOL_VOLUME_STEP);
// Ensure main trade lot is within allowed range
   adjustedLotSize = MathMax(minLot, MathMin(InpLotSize, maxLot));
   adjustedLotSize = NormalizeDouble(MathFloor(adjustedLotSize / stepLot) * stepLot, 2);
   if(adjustedLotSize != InpLotSize)
     {
      if(InpEnableDebug)
         Print("Main LotSize adjusted from ", InpLotSize, " to ", adjustedLotSize,
               " (Min: ", minLot, ", Max: ", maxLot, ", Step: ", stepLot, ")");
     }
// Validate recovery trade lot size
   adjustedLotSizeRecovery = MathMax(minLot, MathMin(InpLotSizeRecovery, maxLot));
   adjustedLotSizeRecovery = NormalizeDouble(MathFloor(adjustedLotSizeRecovery / stepLot) * stepLot, 2);
   if(adjustedLotSizeRecovery != InpLotSizeRecovery)
     {
      if(InpEnableDebug)
         Print("Recovery LotSize adjusted from ", InpLotSizeRecovery, " to ", adjustedLotSizeRecovery,
               " (Min: ", minLot, ", Max: ", maxLot, ", Step: ", stepLot, ")");
     }
// Check if hedging is enabled on the account
   if(AccountInfoInteger(ACCOUNT_MARGIN_MODE) != ACCOUNT_MARGIN_MODE_RETAIL_HEDGING)
     {
      Print("Error: Account does not support hedging. Please enable hedging.");
      return INIT_FAILED;
     }
// Initialize the current trade date
   MqlDateTime time_struct;
   TimeCurrent(time_struct);
   currentTradeDate = StringToTime(StringFormat("%04d.%02d.%02d",
                                   time_struct.year, time_struct.mon, time_struct.day));
// Initialize chart display
//if(ShowInfoOnChart)
//  {
//   CreateInfoObjects();
//   UpdateInfoOnChart();
//  }
// Check if there are already recovery positions open
   int recoveryOpen = CountOpenPositionsMagic(InpMagicRecovery);
   if(recoveryOpen > 0)
     {
      recoveryTradeActive = true;
      recoveryUsedToday = true;
      if(InpEnableDebug)
         Print("Recovery positions detected on startup. Setting recovery flags.");
     }
// Register the stop loss callback

   //ExtTransaction.SetStopLossCallback(OnStopLossHit);



   if(InpEnableDebug)
      Print("MAVERICK EA initialized successfully.");
   PrintTradeStatus();
// Create a trading day line on startup if no positions are open
   if(CountOpenPositionsMagic(InpMagicMain) == 0 && CountOpenPositionsMagic(InpMagicRecovery) == 0)
     {
      if(InpEnableDebug)
         Print("Creating initial trading day line on startup");
      CreateTradingDayLine();
     }
// In OnInit()
   EventSetTimer(UpdateFrequency);
// Initialize the lastClosedPair structure
   lastClosedPair.guid = "";
   lastClosedPair.buyProfit = 0;
   lastClosedPair.sellProfit = 0;
   lastClosedPair.buyHitSL = false;
   lastClosedPair.sellHitSL = false;
   lastClosedPair.totalProfit = 0;
   
   // Initialize the tracked positions array
   ArrayResize(trackedPositions, 0);
   
   return INIT_SUCCEEDED;
  }

//| Expert tick function                                             |
//+------------------------------------------------------------------+
void OnTick()
  {
   static datetime lastUpdateTime = 0;
   datetime currentTimeGUI  = TimeCurrent();
   if(currentTimeGUI - lastUpdateTime >= 1) // Update once per second
     {
      UpdateRunningTotal();
      CheckTrackedPositionsCrossed(); // Add this line to check tracked positions
      lastUpdateTime = currentTimeGUI;
     }
   if(CountOpenPositions() > 0) ManagePositions();
   if(!NewBar())
     {
      // Even when skipping bar processing, update chart info on every tick
      //if(ShowInfoOnChart)
      //   UpdateInfoOnChart();
      return;
     }
   datetime currentTime = TimeCurrent();
   string timeStr = TimeToString(currentTime, TIME_MINUTES);
   CheckNewTradingDay(currentTime);
   if(timeStr == InpTradeTime)
     {
      if(InpEnableDebug)
         Print("Time condition met: current time is ", timeStr);
      int positions = CountOpenPositions();
      if(InpEnableDebug)
         Print("Open positions: ", positions);
      if(InpEnableDebug)
         Print("Daily trading ended flag: ", dailyTradingEnded);
      // Print overall condition result
      if(InpEnableDebug)
         Print("Will open trade: ", (timeStr == InpTradeTime && positions == 0 && !dailyTradingEnded));
     }
// Check if it's time to open the main trade (and no open positions exist)
   if(timeStr == InpTradeTime /*&& CountOpenPositions() == 0 && !dailyTradingEnded*/)
     {
      OpenMainTrade();
      mainTradeOpenTime = currentTime;
      if(InpEnableDebug)
         Print("Main trade opened at ", TimeToString(currentTime));
     }
   DeleteAllLinesIfNoPositions();
   bool foundLine = false;
   for(int i = 0; i < ObjectsTotal(0); i++)
     {
      string objName = ObjectName(0, i);
      if(StringFind(objName, tradingDayLineName) == 0)
        {
         foundLine = true;
         break;
        }
     }
   if(!foundLine && CountOpenPositions() == 0 && !dailyTradingEnded)
     {
      if(InpEnableDebug)
         Print("No trading day line found on chart - creating one");
      CreateTradingDayLine();
     }
  }

void CheckNewTradingDay(datetime currentTime)

     {
      if(InpEnableDebug)
         Print("Day change detected: ", TimeToString(currentTradeDate, TIME_DATE),  " -> ", TimeToString(currentDay, TIME_DATE));
      if(InpEnableDebug)
         Print("BEFORE day change:");

      // Count open positions
      int mainOpen = CountOpenPositionsMagic(InpMagicMain);

            //recoveryUsedToday = true;
            // Reset the breakeven flags here if needed
            BUY_BREAKEVEN_SET = false;
            SELL_BREAKEVEN_SET = false;
            if(InpEnableDebug)

           }
         else
           {

            dailyTradingEnded = false;
            recoveryUsedToday = false;
            // Reset the breakeven flags here if needed
            BUY_BREAKEVEN_SET = false;
            SELL_BREAKEVEN_SET = false;
            if(InpEnableDebug)
               Print("Only main positions open - reset recovery flags for new day");
           }
      // Reset daily profit tracking
      dailyCumulativeProfit = 0.0;
      // Debug output after changes
      if(InpEnableDebug)
         Print("AFTER day change:");
      //PrintTradeStatus();
      // Add this after resetting the flags
      // if(mainOpen == 0 && recoveryOpen == 0)

      CreateTradingDayLine();
      //    Print("TRADING DAY LINE CREATED");

  {
   int count = 0;
   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {

        {
         if(PositionGetString(POSITION_SYMBOL) == Symbol() &&
            (PositionGetInteger(POSITION_MAGIC) == InpMagicMain ||
             PositionGetInteger(POSITION_MAGIC) == InpMagicRecovery))
           {
            count++;
           }
        }
     }
   return count;
  }

//| GenerateTradeGuid: creates a short unique identifier for trades  |
//+------------------------------------------------------------------+
string GenerateTradeGuid()
  {
// Create a simple alphanumeric GUID (10 characters)
   string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
   string guid = "";
// Generate 10 random characters
   for(int i = 0; i < 10; i++)
     {
      int idx = MathRand() % StringLen(chars);
      guid += StringSubstr(chars, idx, 1);
     }
   return guid;
  }

//+------------------------------------------------------------------+
//| OpenMainTrade: opens both Buy and Sell with main inputs          |
//+------------------------------------------------------------------+
void OpenMainTrade()
  {
// Generate a new GUID for this trade pair
   currentTradeGuid = GenerateTradeGuid();
// Create the buy and sell comments with the same GUID
   string buyComment = "B" + currentTradeGuid;
   string sellComment = "S" + currentTradeGuid;
   if(InpEnableDebug)
      Print("Opening main trade pair with GUID: ", currentTradeGuid);
   trade.SetExpertMagicNumber(InpMagicMain);
   BUY_BREAKEVEN_SET = false;
   SELL_BREAKEVEN_SET = false;
// Buy position
   double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
   double slBuy = ask - InpSLPointsMain * Point();
   double tpBuy = ask + InpTPPointsMain * Point();
   if(!trade.Buy(adjustedLotSize, Symbol(), ask, slBuy, tpBuy, buyComment))
     {
      if(InpEnableDebug)
         Print("Error opening main buy position: ", GetLastError());
      return;
     }
// Sell position
   double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
   double slSell = bid + InpSLPointsMain * Point();
   double tpSell = bid - InpTPPointsMain * Point();
   if(!trade.Sell(adjustedLotSize, Symbol(), bid, slSell, tpSell, sellComment))
     {
      if(InpEnableDebug)
         Print("Error opening main sell position: ", GetLastError());
      return;
     }
// Create trigger lines after opening trades
   CreateTrailTriggerLines(ask, bid);
  }

   double slSell = bid + InpSLPointsRecovery * Point();
   double tpSell = bid - InpTPPointsRecovery * Point();
   if(!trade.Sell(adjustedLotSizeRecovery, Symbol(), bid, slSell, tpSell, sellComment))
     {
      if(InpEnableDebug)
         Print("Error opening recovery sell position: ", GetLastError());
      return;
     }
// Mark that we've used recovery today and recovery is active
   recoveryTradeActive = true;
   if(InpEnableDebug)
      Print("Recovery trade opened. No more recovery trades allowed today.");
  }

//+------------------------------------------------------------------+
//| ManagePositions: handles trailing logic & recovery triggers      |
//+------------------------------------------------------------------+
void ManagePositions()
  {
   // if(InpEnableDebug)
   //    Print("=== ManagePositions START ===");
   // if(InpEnableDebug)
   //    Print("Current flags: dailyTradingEnded=", dailyTradingEnded ? "true" : "false",
   //          ", recoveryUsedToday=", recoveryUsedToday ? "true" : "false",
   //          ", recoveryTradeActive=", recoveryTradeActive ? "true" : "false");
// Count positions first
   int mainOpen = CountOpenPositionsMagic(InpMagicMain);
   int recoveryOpen = CountOpenPositionsMagic(InpMagicRecovery);
   // if(InpEnableDebug)
   //    Print("Position counts: mainOpen=", mainOpen, ", recoveryOpen=", recoveryOpen);
//--- First handle RECOVERY TRADE positions closure ---
   if(recoveryOpen == 0 && recoveryTradeActive)
     {
      // if(InpEnableDebug)
      //    Print("DEBUG: Recovery trade closed detection - recoveryOpen=0 but recoveryTradeActive=true");
      // Both recovery positions closed, calculate profit and reset flags
      double totalProfit = GetTotalProfitMagic(InpMagicRecovery);
      dailyCumulativeProfit += totalProfit;
      // if(InpEnableDebug)
      //    Print("DEBUG: Before flag reset - recoveryTradeActive=", recoveryTradeActive ? "true" : "false",
      //          ", recoveryUsedToday=", recoveryUsedToday ? "true" : "false");
      // Reset recovery active flag but KEEP recoveryUsedToday as true
      recoveryTradeActive = false;
      // Always keep recoveryUsedToday as true when a recovery trade closes
      // This ensures no more recovery trades for today, regardless of when it started
      recoveryUsedToday = true;
      // if(InpEnableDebug)
      //    Print("DEBUG: After flag reset - recoveryTradeActive=", recoveryTradeActive ? "true" : "false",
      //          ", recoveryUsedToday=", recoveryUsedToday ? "true" : "false");
      // Set daily trading ended based on profit
      if(totalProfit < 0)
        {
         // if(InpEnableDebug)
         //    Print("DEBUG: Recovery ended in loss, setting dailyTradingEnded=true");
         dailyTradingEnded = true;
         // if(InpEnableDebug)
         //    Print("DEBUG: Recovery ended in loss, setting dailyTradingEnded=true");
         //Print("Recovery trade ended in loss. Trading ended for today with cumulative loss: ", DoubleToString(dailyCumulativeProfit, 2));
        }
      else
        {
         // if(InpEnableDebug)
         //    Print("DEBUG: Recovery ended in profit, setting dailyTradingEnded=true");
         dailyTradingEnded = true;
         // if(InpEnableDebug)
         //    Print("DEBUG: Recovery ended in profit, setting dailyTradingEnded=true");
         //Print("Recovery trade ended in profit. Day complete with cumulative profit: ", DoubleToString(dailyCumulativeProfit, 2));
        }
      // Debug output
      PrintTradeStatus();
     }
   else
      if(recoveryOpen > 0)
        {
         // if(InpEnableDebug)
         //    Print("DEBUG: Recovery trades still open: ", recoveryOpen);
        }
//--- Now handle MAIN TRADE positions ---
   if(mainOpen == 2)
     {
      // if(InpEnableDebug)
      //    Print("DEBUG: Both main positions open, checking for trailing");
      // Both main positions still open (buy & sell) - check for trailing
      ModifyTrailingSL();
     }
   else
      if(mainOpen == 1)
        {
         // if(InpEnableDebug)
         //    Print("DEBUG: One main position closed, moving SL to entry and checking trailing");
         // One main position closed, so move SL of the remaining to entry
         ModifySLToEntry(InpMagicMain);
         TrailSLMainTrade();// Then see if we can trail further
        }
      else
         if(mainOpen == 0 && recoveryTradeActive == false)
           {
            // if(InpEnableDebug)
            //    Print("DEBUG: Both main positions closed and no recovery active, calculating profit");
            // Both main positions closed - check if there's a net loss
            double totalProfit = GetTotalProfitMagic(InpMagicMain);
            dailyCumulativeProfit += totalProfit;
            // Debug output
            // if(InpEnableDebug)
            //    Print("DEBUG: Main trade ended with profit: ", DoubleToString(totalProfit, 2),
            //          " | Recovery used today: ", recoveryUsedToday ? "Yes" : "No",
            //          " | Recovery active: ", recoveryTradeActive ? "Yes" : "No",
            //          " | Daily trading ended: ", dailyTradingEnded ? "Yes" : "No");
            // Remove the totalProfit < 0 check here since recovery trades are handled by OnStopLossHit
            // Just set trading ended flag regardless of profit
            dailyTradingEnded = true;
           // if(InpEnableDebug)
              // Print("DEBUG: Main trade ended. Day complete with profit: ", DoubleToString(dailyCumulativeProfit, 2));
            // Debug output
            PrintTradeStatus();
           }
         else
            if(mainOpen == 0 && recoveryTradeActive == true)
              {
               if(InpEnableDebug)
                  Print("DEBUG: Both main positions closed but recovery is active - no further action needed");
              }
//--- Continue with recovery position management ---
   if(recoveryOpen == 2)
     {
      if(InpEnableDebug)
         Print("DEBUG: Both recovery positions open, applying trailing if configured");
      // Both recovery positions open - apply trailing if configured
      TrailSLRecoveryTrade();
     }
   else
      if(recoveryOpen == 1)
        {
         // if(InpEnableDebug)
         //    Print("DEBUG: One recovery position closed, moving SL of remaining to entry");
         ModifySLToEntry(InpMagicRecovery); // One recovery position closed, move SL of remaining to entry
        }
   // if(InpEnableDebug)
   //    Print("=== ManagePositions END ===");
  }

//+------------------------------------------------------------------+
//| ModifySLToEntry: sets SL = Entry for any remaining position      |
//+------------------------------------------------------------------+
void ModifySLToEntry(int magic)
  {
   // if(InpEnableDebug)
   //   {
   //    Print("_____________________________________");
   //    Print(__FUNCTION__, " magic: ", magic);
   //    Print("_____________________________________");
   //   }
// Find the position that was closed (by looking at history)
   string closedPositionComment = "";
   string oppositePrefix = "";
   if(!HistorySelect(TimeCurrent() - 60, TimeCurrent()))  // Look at last minute
     {
      if(InpEnableDebug)
         Print("Error selecting history");
      return;
     }
   for(int i = HistoryDealsTotal() - 1; i >= 0; i--)
     {
      ulong dealTicket = HistoryDealGetTicket(i);
      if(dealTicket > 0)
        {
         if(HistoryDealGetString(dealTicket, DEAL_SYMBOL) == Symbol() &&
            HistoryDealGetInteger(dealTicket, DEAL_MAGIC) == magic &&
            HistoryDealGetInteger(dealTicket, DEAL_ENTRY) == DEAL_ENTRY_OUT)
           {
            // Get the comment from the closed position
            closedPositionComment = HistoryDealGetString(dealTicket, DEAL_COMMENT);
            // Determine the opposite position prefix
            if(StringSubstr(closedPositionComment, 0, 1) == "B")
               oppositePrefix = "S";
            else
               if(StringSubstr(closedPositionComment, 0, 1) == "S")
                  oppositePrefix = "B";
               else
                  if(StringSubstr(closedPositionComment, 0, 2) == "RB")
                     oppositePrefix = "RS";
                  else
                     if(StringSubstr(closedPositionComment, 0, 2) == "RS")
                        oppositePrefix = "RB";
            // if(InpEnableDebug)
            //   {
            //    Print("Found closed position with comment: ", closedPositionComment,
            //          ", looking for opposite with prefix: ", oppositePrefix);
            //   }
            break;
           }
        }
     }
// If we found a closed position, look for its pair
   string guidPart = "";
   if(closedPositionComment != "")
     {
      // Extract the GUID part
      if(StringSubstr(closedPositionComment, 0, 1) == "B" ||
         StringSubstr(closedPositionComment, 0, 1) == "S")
        {
         guidPart = StringSubstr(closedPositionComment, 1);
        }
      else
         if(StringSubstr(closedPositionComment, 0, 2) == "RB" ||
            StringSubstr(closedPositionComment, 0, 2) == "RS")
           {
            guidPart = StringSubstr(closedPositionComment, 2);
           }
      // if(InpEnableDebug)
      //    Print("Extracted GUID part: ", guidPart);
     }
// Now find and modify the opposite position
   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(PositionSelectByTicket(PositionGetTicket(i)))
        {
         if(PositionGetString(POSITION_SYMBOL) == Symbol() &&
            PositionGetInteger(POSITION_MAGIC) == magic)
           {
            string posComment = PositionGetString(POSITION_COMMENT);
            // If we have a GUID to match, check if this is the right position
            if(guidPart != "" && oppositePrefix != "")
              {
               // Check if this position has the opposite prefix and same GUID
               bool isMatch = false;
               if(oppositePrefix == "B" || oppositePrefix == "S")
                 {
                  isMatch = (StringSubstr(posComment, 0, 1) == oppositePrefix &&
                             StringSubstr(posComment, 1) == guidPart);
                 }
               else // RB or RS
                 {
                  isMatch = (StringSubstr(posComment, 0, 2) == oppositePrefix &&
                             StringSubstr(posComment, 2) == guidPart);
                 }
               if(!isMatch)
                 {
                  if(InpEnableDebug)
                     Print("Position ", PositionGetTicket(i), " with comment ", posComment,
                           " does not match the opposite we're looking for");
                  continue;
                 }
               if(InpEnableDebug)
                  Print("Found matching opposite position: ", PositionGetTicket(i),                     " with comment: ", posComment);
              }
            // Check if breakeven has already been set for this position type
            if((PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY && BUY_BREAKEVEN_SET) ||
               (PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL && SELL_BREAKEVEN_SET))
              {
               continue;
              }
            double entry = PositionGetDouble(POSITION_PRICE_OPEN);
            double currentSL = PositionGetDouble(POSITION_SL);
            // If the SL is not already at entry, move it
            if(MathAbs(currentSL - entry) > Point())
              {
               if(!trade.PositionModify(PositionGetTicket(i), entry,
                                        PositionGetDouble(POSITION_TP)))
                 {
                  if(InpEnableDebug)
                     Print("Error modifying SL to entry for position ",
                           PositionGetTicket(i), ": ", GetLastError());
                 }
               else
                 {
                  if(InpEnableDebug)
                     Print(__FUNCTION__);
                  if(InpEnableDebug)
                     Print("Breakeven on position ", PositionGetTicket(i));
                  // Set the appropriate breakeven flag
                  if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
                     BUY_BREAKEVEN_SET = true;
                  else
                     if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL)
                        SELL_BREAKEVEN_SET = true;
                  // Display breakeven notification on chart
                  //if(ShowInfoOnChart)
                  //   Comment("Breakeven adjustment made at ", TimeToString(TimeCurrent()));
                  // Update chart info
                  //if(ShowInfoOnChart)
                  //   UpdateInfoOnChart();
                 }
              }
            // If we found a specific match, we can stop after modifying it
            if(guidPart != "" && oppositePrefix != "")
               break;
           }
        }
     }
  }

//+------------------------------------------------------------------+
//| TrailSLMainTrade: second trailing after a bigger move           |
//+------------------------------------------------------------------+
void TrailSLMainTrade()
  {
   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(PositionSelectByTicket(PositionGetTicket(i)))
        {
         if(PositionGetString(POSITION_SYMBOL) == Symbol() &&
            PositionGetInteger(POSITION_MAGIC) == InpMagicMain)
           {
            double entry     = PositionGetDouble(POSITION_PRICE_OPEN);
            double currentSL = PositionGetDouble(POSITION_SL);
            double profitPoints = 0.0;
            if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
              {
               double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
               profitPoints = (bid - entry) / Point();
               // Skip if SL already above entry
               if(currentSL > entry)
                 {
                  continue;
                 }
              }
            else
               if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL)
                 {
                  double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
                  profitPoints = (entry - ask) / Point();
                  // Skip if SL already below entry
                  if(currentSL < entry)
                    {
                     continue;
                    }
                 }
            // Check if SL is at entry AND profit >= InpTrail2TriggerMain => move SL by InpTrail2OffsetMain
            if(MathAbs(currentSL - entry) <= Point() && profitPoints >= InpTrail2TriggerMain)
              {
               double newSL;
               if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY && currentSL <= entry)
                  newSL = entry + InpTrail2OffsetMain * Point();
               else
                  if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL && currentSL >= entry)
                     newSL = entry - InpTrail2OffsetMain * Point();
               if(!trade.PositionModify(PositionGetTicket(i), newSL, PositionGetDouble(POSITION_TP)))
                 {
                  if(InpEnableDebug)
                     Print("Error trailing SL for position ", PositionGetTicket(i), ": ", GetLastError());
                 }
               else
                 {
                  if(InpEnableDebug)
                     Print("SL trailed to ", newSL, " for position ", PositionGetTicket(i));
                  
                  // Track this position with trailed SL
                  ulong ticket = PositionGetTicket(i);
                  string comment = PositionGetString(POSITION_COMMENT);
                  string guid = "";
                  
                  // Extract GUID from position comment
                  if(StringSubstr(comment, 0, 1) == "B")
                     guid = StringSubstr(comment, 1);
                  else if(StringSubstr(comment, 0, 1) == "S")
                     guid = StringSubstr(comment, 1);
                  
                  if(guid != "")
                  {
                     // Check if position is already tracked
                     bool positionFound = false;
                     for(int j = 0; j < ArraySize(trackedPositions); j++)
                     {
                        if(trackedPositions[j].ticket == ticket)
                        {
                           trackedPositions[j].trailedSL = newSL;
                           positionFound = true;
            break;
           }
                     }
                     
                     // If not tracked, add to array
                     if(!positionFound)
                     {
                        int size = ArraySize(trackedPositions);
                        ArrayResize(trackedPositions, size + 1);
                        trackedPositions[size].ticket = ticket;
                        trackedPositions[size].guid = guid;
                        trackedPositions[size].trailedSL = newSL;
                        trackedPositions[size].slTriggered = false;
                        trackedPositions[size].blockRecovery = false;
                        
                  if(InpEnableDebug)
                           Print("Position ", ticket, " with GUID ", guid, " tracked with trailed SL at ", newSL);
                     }
                  }
                 }
              }
           }
        }
     }
  }

//+------------------------------------------------------------------+
//| TrailSLRecoveryTrade: moves SL to entry when profit reached      |
//+------------------------------------------------------------------+
void TrailSLRecoveryTrade()
  {
   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(PositionSelectByTicket(PositionGetTicket(i)))
        {
         if(PositionGetString(POSITION_SYMBOL) == Symbol() &&
            PositionGetInteger(POSITION_MAGIC) == InpMagicRecovery)
           {
            double entry     = PositionGetDouble(POSITION_PRICE_OPEN);
            double currentSL = PositionGetDouble(POSITION_SL);
            double profitPoints = 0.0;
            // Skip if SL already modified (at entry or better)
            if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY && currentSL >= entry)
              {
               continue;
              }
            else
               if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL && currentSL <= entry)
                 {
                  continue;
                 }
            // Calculate profit in points
            if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
              {
               double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
               profitPoints = (bid - entry) / Point();
              }
            else
               if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL)
                 {
                  double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
                  profitPoints = (entry - ask) / Point();
                 }
            // Check if profit >= InpTrail2TriggerRecovery => move SL to entry
            if(profitPoints >= InpTrail2TriggerRecovery)
              {
               if(!trade.PositionModify(PositionGetTicket(i), entry, PositionGetDouble(POSITION_TP)))
                 {
                  if(InpEnableDebug)
                     Print("Error moving recovery SL to entry for position ", PositionGetTicket(i), ": ", GetLastError());
                 }
               else
                 {
                  if(InpEnableDebug)
                     Print("Recovery SL moved to entry for position ", PositionGetTicket(i),
                           " after ", profitPoints, " points profit");
                  // Update chart info
                  //if(ShowInfoOnChart)
                  //  {
                  //   Comment("Recovery trade SL moved to entry at ", TimeToString(TimeCurrent()));
                  //   UpdateInfoOnChart();
                  //  }
                 }
              }
           }
        }
     }
  }

//| DeleteAllLinesIfNoPositions: removes HLines when no trades open  |
//+------------------------------------------------------------------+
void DeleteAllLinesIfNoPositions()
  {
// Check if there are no open positions for this symbol with our magic numbers
   if(CountOpenPositions() == 0)
     {
      // Delete the trigger lines
      DeleteTrailTriggerLine(infoLabelPrefix + "BuyTrailTrigger");
      DeleteTrailTriggerLine(infoLabelPrefix + "SellTrailTrigger");
      // Delete all trading day lines older than today
      MqlDateTime today;
      TimeCurrent(today);
      datetime todayStart = StringToTime(StringFormat("%04d.%02d.%02d 00:00",
                                         today.year, today.mon, today.day));
      // Iterate through all objects
      for(int i = ObjectsTotal(0) - 1; i >= 0; i--)
        {
         string objName = ObjectName(0, i);
         if(StringFind(objName, tradingDayLineName) == 0)  // If object name starts with our prefix
           {
            datetime lineTime = (datetime)ObjectGetInteger(0, objName, OBJPROP_TIME);
            if(lineTime < todayStart)  // If line is from previous day
              {
               ObjectDelete(0, objName);
              }
           }
        }
     }
  }

//| IsSameDay: Checks if two datetime values are on the same day     |
//+------------------------------------------------------------------+
bool IsSameDay(datetime time1, datetime time2)
  {
   MqlDateTime time_struct1, time_struct2;
   TimeToStruct(time1, time_struct1);
   TimeToStruct(time2, time_struct2);
   return (time_struct1.year == time_struct2.year &&
           time_struct1.mon == time_struct2.mon &&
           time_struct1.day == time_struct2.day);
  }

// Add this new function to check if price has crossed the trailed SL level
void CheckTrackedPositionsCrossed()
{
   for(int i = 0; i < ArraySize(trackedPositions); i++)
   {
      // Skip positions that have already triggered their SL
      if(trackedPositions[i].slTriggered)
         continue;
      
      // Check if position is still open
      if(PositionSelectByTicket(trackedPositions[i].ticket))
      {
         // Position is still open - check if price has crossed back over entry
         double entryPrice = PositionGetDouble(POSITION_PRICE_OPEN);
         double currentPrice;
         bool priceReturnedToEntry = false;
         
         // Get current market price based on position type
         if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
         {
            currentPrice = SymbolInfoDouble(Symbol(), SYMBOL_BID);
            // For buy positions, check if price went below entry and then back above
            if(currentPrice >= entryPrice && trackedPositions[i].priceBelowEntry)
            {
               priceReturnedToEntry = true;
               if(InpEnableDebug)
                  Print("Buy position ", trackedPositions[i].ticket, " price returned above entry after going below");
            }
            else if(currentPrice < entryPrice)
            {
               // Mark that price has been below entry
               trackedPositions[i].priceBelowEntry = true;
            }
         }
         else // SELL position
         {
            currentPrice = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
            // For sell positions, check if price went above entry and then back below
            if(currentPrice <= entryPrice && trackedPositions[i].priceAboveEntry)
            {
               priceReturnedToEntry = true;
               if(InpEnableDebug)
                  Print("Sell position ", trackedPositions[i].ticket, " price returned below entry after going above");
            }
            else if(currentPrice > entryPrice)
            {
               // Mark that price has been above entry
               trackedPositions[i].priceAboveEntry = true;
            }
         }
         
         // If price has returned to entry after crossing it, open a recovery hedge pair
         if(priceReturnedToEntry && !trackedPositions[i].recoveryOpened)
         {
            if(InpEnableDebug)
               Print("Opening recovery hedge pair for position ", trackedPositions[i].ticket, 
                     " with GUID ", trackedPositions[i].guid, " as price returned to entry");
            
            // Mark this position as having opened a recovery so we don't do it again
            trackedPositions[i].recoveryOpened = true;
            
            // Call OpenRecoveryTrade to open a new hedge pair
            OpenRecoveryTrade();
         }
      }
      else
      {
         // Position closed, check if it was due to SL hit
         bool slTriggered = false;
         
         // We need to check position history to see if it was closed at the SL level
         if(HistorySelectByPosition(trackedPositions[i].ticket))
         {
            for(int j = 0; j < HistoryDealsTotal(); j++)
            {
               ulong dealTicket = HistoryDealGetTicket(j);
               if(HistoryDealGetInteger(dealTicket, DEAL_ENTRY) == DEAL_ENTRY_OUT &&
                  HistoryDealGetInteger(dealTicket, DEAL_REASON) == DEAL_REASON_SL)
               {
                  slTriggered = true;
                  break;
               }
            }
         }
         
         if(slTriggered)
         {
            trackedPositions[i].slTriggered = true;
            trackedPositions[i].blockRecovery = true;
            
            skipRecovery = true;
            
            // Block recovery for all positions with the same GUID
            string currentGuid = trackedPositions[i].guid;
            for(int j = 0; j < ArraySize(trackedPositions); j++)
            {
               if(trackedPositions[j].guid == currentGuid)
                  trackedPositions[j].blockRecovery = true;
            }
            
            if(InpEnableDebug)
               Print("Position ", trackedPositions[i].ticket, " with GUID ", trackedPositions[i].guid, 
                     " hit trailed SL - recovery blocked for this hedge set");
         }
      }
   }
}

//| CreateTrailTriggerLines: creates dotted gray lines at trigger    |
//+------------------------------------------------------------------+
void CreateTrailTriggerLines(double buyEntry, double sellEntry)
  {
// Buy trigger line
   string buyLineName = infoLabelPrefix + "BuyTrailTrigger";
   double buyTriggerPrice = buyEntry + InpTrail2TriggerMain * Point();
   ObjectCreate(0, buyLineName, OBJ_HLINE, 0, 0, buyTriggerPrice);
   ObjectSetInteger(0, buyLineName, OBJPROP_COLOR, clrGray);
   ObjectSetInteger(0, buyLineName, OBJPROP_STYLE, STYLE_DOT);
   ObjectSetInteger(0, buyLineName, OBJPROP_WIDTH, 1);
// Sell trigger line
   string sellLineName = infoLabelPrefix + "SellTrailTrigger";
   double sellTriggerPrice = sellEntry - InpTrail2TriggerMain * Point();
   ObjectCreate(0, sellLineName, OBJ_HLINE, 0, 0, sellTriggerPrice);
   ObjectSetInteger(0, sellLineName, OBJPROP_COLOR, clrGray);
   ObjectSetInteger(0, sellLineName, OBJPROP_STYLE, STYLE_DOT);
   ObjectSetInteger(0, sellLineName, OBJPROP_WIDTH, 1);
  }

//| GetTotalProfitMagic: sums closed profit for a specific magic     |
//+------------------------------------------------------------------+
double GetTotalProfitMagic(int magic)
  {
   double totalProfit = 0.0;
   // We only count deals from the time we opened the main trade
   if(!HistorySelect(mainTradeOpenTime, TimeCurrent()))
     {
      if(InpEnableDebug)
         Print("Error selecting history: ", GetLastError());
      return 0;
     }
   // Sum all profits from closed positions with the specified magic number
   for(int i = HistoryDealsTotal() - 1; i >= 0; i--)
     {
      ulong dealTicket = HistoryDealGetTicket(i);
      if(dealTicket > 0)
        {
         if(HistoryDealGetString(dealTicket, DEAL_SYMBOL) == Symbol() &&
            HistoryDealGetInteger(dealTicket, DEAL_MAGIC) == magic &&
            HistoryDealGetInteger(dealTicket, DEAL_ENTRY) == DEAL_ENTRY_OUT)
           {
            totalProfit += HistoryDealGetDouble(dealTicket, DEAL_PROFIT);
           }
        }
     }
   return totalProfit;
  }

//| ModifyTrailingSL: Checks price movement and moves SL accordingly |
//+------------------------------------------------------------------+
void ModifyTrailingSL()
  {
   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(PositionSelectByTicket(PositionGetTicket(i)))
        {
         if(PositionGetString(POSITION_SYMBOL) == Symbol() &&
            PositionGetInteger(POSITION_MAGIC) == InpMagicMain)
           {
            double entry = PositionGetDouble(POSITION_PRICE_OPEN);
            double currentSL = PositionGetDouble(POSITION_SL);
            // Skip if SL is already moved (not at initial SL placement)
            if(MathAbs(currentSL - entry) < InpSLPointsMain * Point() * 0.9)
               continue;
            bool moveSL = false;
            // For BUY positions - check if price moved up by InpTrail2OffsetMain points
            if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
              {
               double bid = SymbolInfoDouble(Symbol(), SYMBOL_BID);
               double priceMovement = bid - entry;
               // If price moved up by the offset amount, move SL to entry
               if(priceMovement >= InpTrail2OffsetMain * Point())
                 {
                  moveSL = true;
                  if(InpEnableDebug)
                     Print("Buy price moved up ", priceMovement / Point(), " points - moving SL to entry");
                 }
              }
            // For SELL positions - check if price moved down by InpTrail2OffsetMain points
            else
               if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL)
                 {
                  double ask = SymbolInfoDouble(Symbol(), SYMBOL_ASK);
                  double priceMovement = entry - ask;
                  // If price moved down by the offset amount, move SL to entry
                  if(priceMovement >= InpTrail2OffsetMain * Point())
                    {
                     moveSL = true;
                     if(InpEnableDebug)
                        Print("Sell price moved down ", priceMovement / Point(), " points - moving SL to entry");
                    }
                 }
            // If conditions met, modify SL to entry
            if(moveSL)
              {
               if(!trade.PositionModify(PositionGetTicket(i), entry, PositionGetDouble(POSITION_TP)))
                 {
                  if(InpEnableDebug)
                     Print("Error modifying SL to entry for position ",
                           PositionGetTicket(i), ": ", GetLastError());
                 }
               else
                 {
                  if(InpEnableDebug)
                     Print(__FUNCTION__);
                  if(InpEnableDebug)
                     Print("Breakeven on position ", PositionGetTicket(i));
                  // Set the appropriate breakeven flag
                  if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
                     BUY_BREAKEVEN_SET = true;
                  else
                     if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL)
                        SELL_BREAKEVEN_SET = true;
                 }
              }
           }
        }
     }
  }
