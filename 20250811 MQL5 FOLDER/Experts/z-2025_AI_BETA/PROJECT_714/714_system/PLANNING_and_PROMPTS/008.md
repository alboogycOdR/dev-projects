//+------------------------------------------------------------------+
//| Check for Bullish Order Block                                    |
//| Adapted to be called from the EA with price arrays and index   |
//| Looks at index candle as potential OB and checks bars before for move|
//+------------------------------------------------------------------+
bool IsBullishOrderBlock(int index, const double &open[], const double &high[], const double &low[], const double &close[], int rates_total)
{
    // Check index boundaries
    if (index < ob_Lookback_Bars + ob_MaxBlockCandles - 1 || index >= rates_total) return false;

    // Look for last bearish candle before a strong bullish move
    if(close[index] < open[index]) // Check if candle at 'index' is bearish (the potential OB candle)
    {
        double move = 0;
        int bullishCandles = 0;
        // Scan *after* the potential OB candle (index+1, index+2, ...) for the bullish move
        for(int i = index + 1; i < index + 1 + ob_Lookback_Bars && i < rates_total; i++) // Check forward after the OB candidate candle
        {
             // Check if candle 'i' is bullish
             if(close[i] > open[i])
             {
                move += (close[i] - open[i]) / GetPipValue();
                bullishCandles++;
                // Check if the cumulative bullish move and number of bullish candles meet the criteria
                if(move >= ob_MinMovePips && bullishCandles >= 2)
                {
                     // Ensure no higher high *in the block candles AFTER the potential OB* (candle at index is the OB)
                     // The indicator's check for higher high in the block seems reversed - checking ahead, not behind
                     // Assuming the intended logic is no higher high in the candles *that form the OB*
                     // The indicator checked `high[j] > high[index]` for j from index to index+MaxBlockCandles.
                     // A common definition is that the high of the OB candle should not be exceeded by subsequent *few* candles if looking for retest
                     // Let's implement the indicator's apparent logic as it was provided. Check highs from the OB candle (index) onwards for MaxBlockCandles.
                    bool no_higher_high_in_block_period = true;
                    for(int j = index; j < index + ob_MaxBlockCandles && j < rates_total; j++)
                    {
                       if (j > index && high[j] > high[index]) { // Check highs of subsequent candles against the potential OB candle's high
                         no_higher_high_in_block_period = false;
                         break;
                       }
                    }
                    if (no_higher_high_in_block_period)
                    {
                         return true; // Found a valid Bullish Order Block based on this logic
                    }
                }
             }
             else break; // Stop if a non-bullish candle is found immediately after the block
        }
    }
    return false;
}

//+------------------------------------------------------------------+
//| Function to perform daily calculation and visualize timings      |
//+------------------------------------------------------------------+
void CalculateAndDrawDailyTimings()
{
    // --- Calculate today's Primary Key Time (13:00 UTC+2) in server time ---
    datetime server_midnight = iTime(Symbol(), PERIOD_D1, 0);
    if(server_midnight == 0) {
        Print("Error getting server midnight time.");
        g_last_initialized_day_time = 0; // Mark initialization failed for the day
        return; // Cannot calculate without this
    }

    // Target hour in GMT (13:00 UTC+2 is 11:00 GMT -> 13 - 2 = 11)
    int target_hour_gmt = utcPlus2_KeyHour - 2; // Assuming UTC+2 is always GMT+2 (needs confirmation with real standard)

    // Calculate target hour in Server Time relative to server midnight
    int target_server_hour = target_hour_gmt + server_GMT_Offset_Manual;

    // Ensure target_server_hour is within 0-23
    target_server_hour = target_server_hour % 24;
    if(target_server_hour < 0) target_server_hour += 24;

    // Calculate the exact datetime in server time for the key time
    g_TodayKeyTime_Server = server_midnight + (long)(target_server_hour * 3600 + utcPlus2_KeyMinute * 60);


    //--- Calculate the observation end time (after observation duration) in server time ---
    g_ObservationEndTime_Server = g_TodayKeyTime_Server + observation_Duration_Minutes * 60;

     Print("Calculated Daily Timings for ", TimeToString(server_midnight, TIME_DATE));
     Print(" Key Time (Server): ", TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES));
     Print(" Obs End (Server): ", TimeToString(g_ObservationEndTime_Server, TIME_DATE|TIME_MINUTES));


    //--- Calculate the specific 15th candlestick entry time in server time ---
    // Find the bar index that starts exactly at or immediately after the Key Time (g_TodayKeyTime_Server)
    // We want the 15th M5 bar starting *from* the bar at g_TodayKeyTime_Server as the first bar.
    int key_bar_idx_for_count = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);

    if (key_bar_idx_for_count >= 0) // Ensure the starting bar for the count exists
    {
        // The index of the target bar relative to chart start is Key bar index + (Target count - 1)
        int target_entry_bar_idx = key_bar_idx_for_count + (entry_Candlestick_Index -1);

        // Get the time of the target entry bar (assuming enough bars exist)
        if(target_entry_bar_idx >= 0 && target_entry_bar_idx < Bars(Symbol(), Period()))
        {
             g_EntryTiming_Server = iTime(Symbol(), Period(), target_entry_bar_idx);
             Print(visual_comment_text + " - Calculated Target 15th M5 Candlestick Entry Time (Server Time): ", TimeToString(g_EntryTiming_Server, TIME_DATE|TIME_MINUTES));
        }
         else
         {
              Print("Warning: Not enough historical bars to determine the 15th candlestick time based on today's key time. (", target_entry_bar_idx, "/", Bars(Symbol(),Period()), ")");
              g_EntryTiming_Server = 0; // Indicate timing not calculated
         }

         // --- Get the price at the Key Time bar for reference ---
         // Find the bar index at the exact Key Time (g_TodayKeyTime_Server) or the first one after
         int key_price_bar_index = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);
         if(key_price_bar_index >= 0 && key_price_bar_index < Bars(Symbol(), Period()))
         {
              g_KeyPrice_At_KeyTime = iOpen(Symbol(), Period(), key_price_bar_index); // Use Open price of the bar at the Key Time
              Print(visual_comment_text + " - Price at Key Time bar open: ", DoubleToString(g_KeyPrice_At_KeyTime, Digits()));
         }
         else
         {
              Print("Warning: Could not find a bar at or after the calculated Key Time to get the price.");
              g_KeyPrice_At_KeyTime = 0;
         }

    } else {
         Print("Warning: Could not find a bar at or after the calculated Key Time (", TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES), ") to start the 15-bar count. Insufficient data?");
         g_EntryTiming_Server = 0;
         g_KeyPrice_At_KeyTime = 0;
    }

    //--- Draw Visual Lines ---
    if (visual_lines_enabled)
    {
        // Ensure previous lines are removed before drawing new ones for the day
        RemoveDailyVisualObjects(); // Removes previous timing and OB visuals

        // Draw Vertical Line at Key Time (13:00 UTC+2 Server Time)
        if (g_TodayKeyTime_Server > 0 && ObjectCreate(0, "VLine_"+IntegerToString(TimeTradeServer()), OBJ_VLINE, 0, g_TodayKeyTime_Server)) // Using unique name based on initialization time to avoid conflict on redraw
        {
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_COLOR, vline_keytime_color); // Use ObjectName(0,0) after Create to get the name
            ObjectSetString(0, ObjectName(0,0), OBJPROP_TEXT, visual_comment_text + ": 13:00 UTC+2 (" + TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES) + " Server)");
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_RAY, false);
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_SELECTABLE, false);
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
        }

        // Draw Vertical Line at Observation End Time (Server Time)
        if (g_ObservationEndTime_Server > 0 && ObjectCreate(0, "VLineObsEnd_"+IntegerToString(TimeTradeServer()), OBJ_VLINE, 0, g_ObservationEndTime_Server))
        {
             ObjectSetInteger(0, ObjectName(0,0), OBJPROP_COLOR, vline_obsend_color);
            ObjectSetString(0, ObjectName(0,0), OBJPROP_TEXT, visual_comment_text + ": Obs End (" + TimeToString(g_ObservationEndTime_Server, TIME_DATE|TIME_MINUTES) + " Server)");
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_RAY, false);
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_SELECTABLE, false);
             ObjectSetInteger(0, ObjectName(0,0), OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
        }
         
        // Draw Optional Horizontal Line at Key Price
        if (visual_key_price_line && g_KeyPrice_At_KeyTime > 0 && ObjectCreate(0, "HLineKeyPrice_"+IntegerToString(TimeTradeServer()), OBJ_HLINE, 0, g_KeyPrice_At_KeyTime))
        {
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_COLOR, hline_keyprice_color);
            ObjectSetString(0, ObjectName(0,0), OBJPROP_TEXT, visual_comment_text + ": Key Price");
            ObjectSetInteger(0, ObjectName(0,0), OBJPROP_SELECTABLE, false);
        }
    }

     // --- Update the last initialized day marker ---
    g_last_initialized_day_time = server_midnight;

     // --- Reset static counters/arrays for daily OBs ---
     static st_OrderBlock dailyOrderBlocks[100]; // Redefine or access, MQL5 handles static initialization per EA instance

}

    }

**Key Changes in Version 2:**

1.  **Copied OB Detection Logic:** The `IsBullishOrderBlock` and `IsBearishOrderBlock` functions (adapted to take price arrays and index) are included directly in the EA code. A helper `GetPipValue` is added for pip calculations.
2.  **OB Detection Triggered Daily:** A new boolean flag `g_order_blocks_detected_today` ensures that a scan for Order Blocks is attempted only *once* each day after the initial bias is determined.
3.  **`DetectAndStoreOrderBlocks` Function:** This new function is called after bias determination. It defines a historical range (from Observation End back a certain number of bars) to scan for potential Order Blocks using the copied `Is...OrderBlock` logic. It uses simple `CopyBuffer` equivalents (`CopyOpen`, `CopyHigh`, etc.) to get the price data for scanning.
4.  **Basic OB Storage & Visualization (V2 Simplification):** Instead of a complex `CArrayObj` in this core version, I've used a `static st_OrderBlock dailyOrderBlocks` and `static int ob_count = 0` to store the *details* of the first 100 detected Order Blocks for the day. This simplifies the code structure in V2. If `visual_order_blocks` is true, it draws rectangle objects and labels on the chart for the detected OBs, naming them with a standard prefix "OB\_".
5.  **`UpdateMitigationStatus`:** A basic function is added to check if detected and stored Order Blocks have been "mitigated" (price has moved through their range). It marks them as mitigated in the array and optionally changes the color of the visual object to gray.
6.  **Entry Trigger Updates:** The `OnTimer` logic now specifically looks for the `g_EntryTiming_Server` (the 15th candlestick time). The placeholder comments in `YourBuyEntryConditionsMet` and `YourSellEntryConditionsMet` are updated to indicate that you need to iterate through the `dailyOrderBlocks` array *within these functions* to check if the price on the current bar (`closed_bar_index`) interacts with any relevant, unmitigated OB, and whether your additional confirmation (price action, FVG - needs your code) is met *at that time*.
7.  **Trade Parameter Placeholders:** Inputs for placeholder SL and TP distances in pips (`stop_Loss_Pips_Placeholder`, `take_Profit_Pips_Placeholder`) are added, along with placeholder functions for `CalculateLotSize`, `PlaceBuyOrder`, and `PlaceSellOrder`. These placeholders in the comments explain what needs to be implemented there based on the full strategy rules (e.g., calculating lot size based on risk, setting SL/TP based on detected levels).
8.  **Refined Visual Object Naming and Removal:** Vertical/Horizontal lines now use a name incorporating `TimeTradeServer()` for better uniqueness on redraws within a day. `RemoveDailyVisualObjects` is updated to try and remove objects with the specific naming prefixes used by the EA ("VLine\_", "HLine\_", "OB\_").
