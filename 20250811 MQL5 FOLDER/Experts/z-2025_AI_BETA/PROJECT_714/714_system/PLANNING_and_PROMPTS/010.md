// Helper function to calculate pip value - already in V2 logic, moved here for clarity
double GetPipValue()
{
    double pip_value = Point();
    if(Digits() == 3 || Digits() == 5) pip_value *= 10;
    return pip_value;
}

// Check if a candle at 'index' (in array) is a Bullish Order Block candidate
// (Bearish candle followed by consecutive bullish candles moving a minimum distance)
bool IsBullishOrderBlockCandidate(int index, const double &open[], const double &high[], const double &low[], const double &close[], int array_size)
{
    // Candle at 'index' must be bearish
    if (close[index] >= open[index]) return false;

    // Need at least 2 candles after the candidate OB for the impulsive move check
    if (index + 1 + 2 > array_size) return false; // index + 1 is first candle after OB

    double cumulative_move_after_ob = 0;
    int bullish_candles_count = 0;
    bool found_strong_move = false;

    // Scan *after* the potential OB candle (index+1, index+2, ...) for the bullish move
    // The scan range after the OB should probably relate to how soon the "impulse" happens after Stage 2
    // Let's use ob_Lookback_Bars_For_Impulse to limit how far after the OB candle we look for the confirming move.
    for (int i = index + 1; i < index + 1 + ob_Lookback_Bars_For_Impulse && i < array_size; i++)
    {
         if (close[i] > open[i]) // Found a bullish candle
         {
              cumulative_move_after_ob += (close[i] - open[i]); // Add the points move
              bullish_candles_count++;

              // Check for the minimum cumulative move AND minimum number of confirming bullish candles
              if (cumulative_move_after_ob / GetPipValue() >= ob_MinMovePips && bullish_candles_count >= 2) // >= 2 consecutive bullish candles summing to min pips move
              {
                   found_strong_move = true; // Criteria for strong move met in this direction
                   break; // Stop scanning in this direction, impulse found
              }
         }
         // If a bearish or doji candle breaks the sequence of bullish candles before move criteria met,
         // the "consecutive bullish candles" idea might fail depending on exact interpretation.
         // The previous indicator's logic `else break;` implied a break in the *type* of candle stops the impulse count. Let's follow that.
         else
         {
              // If the move criteria was not met with consecutive candles, this candle stops the check for this potential OB
              break;
         }
    }

     if (!found_strong_move) return false; // No valid impulsive move after the candidate OB

    // Ensure no high *in the potential OB candle and next MaxBlockCandles bars* is higher than the OB candle's high
    // This check helps define the block itself and confirms it's the origin point that wasn't immediately surpassed.
    bool no_higher_high_in_block_period = true;
    // Index 'index' is the OB candle. Loop from 'index' up to 'index + ob_MaxBlockCandles - 1'
    for(int j = index; j < index + ob_MaxBlockCandles && j < array_size; j++)
    {
        // If we are checking a candle *after* the potential OB candle, ensure its high is not greater.
        if (j > index && high[j] > high[index])
        {
           no_higher_high_in_block_period = false;
           break;
        }
        // Also, if the potential OB candle itself (index) had a high equal to or very close to a strong prior swing high (for bearish OB - simplified concept)
        // or a low equal to a strong prior swing low (for bullish OB), it might make it a more relevant OB.
        // This logic is not in the provided snippets but is common in some SMC definitions. Let's stick to provided.
    }
    if (!no_higher_high_in_block_period) return false;


   // If all checks passed, this is a Bullish Order Block
   return true;
}

//| Detects and Stores Order Blocks formed around the Key Time       |
//| Runs once per day after initial bias determination               |
//| Scan range limited to the expected manipulation period (Stage 2) |
//+------------------------------------------------------------------+
void DetectOrderBlocksForToday(int scan_from_index, int scan_to_index)
{
     if (g_order_blocks_scanned_today) return; // Only run detection scan once per day

     int total_bars = Bars(Symbol(), Period());

     // Define a smaller, relevant scan range based on the strategy timing.
     // We are looking for OBs formed *during* or *just before* the initial manipulation phase (Stage 2)
     // The manipulation happens *after* g_TodayKeyTime_Server and typically before or around g_ObservationEndTime_Server.
     // A reasonable scan range is from the Observation End time backwards to the Key Time plus some lookback for OB logic
     int scan_end_bar_idx_recent = iBarShift(Symbol(), Period(), g_ObservationEndTime_Server + PeriodSeconds(Period()), false); // Scan up to this point (more recent)
     int scan_start_bar_idx_oldest = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false); // Scan from around the Key Time

     if(scan_end_bar_idx_recent < 0 || scan_start_bar_idx_oldest < 0 || scan_end_bar_idx_recent >= total_bars)
     {
          Print(visual_comment_text + " - Cannot determine valid scan range for Order Blocks. Insufficient data around key times.");
          g_order_blocks_scanned_today = true; // Avoid trying again today
          return;
     }

     // Ensure the scan range includes enough historical bars for the OB detection lookback (`ob_Lookback_Bars_For_Impulse` etc.)
     int earliest_bar_needed = scan_start_bar_idx_oldest + ob_Lookback_Bars_For_Impulse + ob_MaxBlockCandles;
     if (earliest_bar_needed >= total_bars)
     {
         Print(visual_comment_text + " - Not enough historical bars available to perform Order Block scan in the relevant period.");
          g_order_blocks_scanned_today = true;
         return;
     }

     // Calculate the actual number of bars we need to copy data for to cover the scan range plus lookback needed for each scan bar
     // Scan from `scan_end_bar_idx_recent` back to `scan_start_bar_idx_oldest`. For each bar `i` in this range, the OB check
     // potentially looks back `ob_Lookback_Bars_For_Impulse` bars, but our implementation looks *forward* for the impulse move.
     // The candidate OB is at index `i`, the impulse check is `i+1` onwards for `ob_Lookback_Bars_For_Impulse` bars.
     // The block definition looks from `i` up to `i + ob_MaxBlockCandles - 1`.
     // The range to copy data for should be from the earliest possible index required by the OB logic (`scan_start_bar_idx_oldest`)
     // to the latest bar we might scan *for a candidate OB* (`scan_end_bar_idx_recent`).
     int copy_from_index = scan_start_bar_idx_oldest - ob_Lookback_Bars_For_Impulse - ob_MaxBlockCandles; // Needs lookback for checks starting at scan_start_bar_idx_oldest
     if(copy_from_index < 0) copy_from_index = 0;
     int copy_to_index = total_bars -1 ; // Copy up to the latest available bar to handle checks forward

     // Total bars to copy
     int num_bars_to_copy = total_bars - copy_from_index;

//+------------------------------------------------------------------+
//| Function to perform daily calculation and visualize timings      |
//+------------------------------------------------------------------+
void CalculateAndDrawDailyTimings()
{
    // --- Calculate today's Primary Key Time (13:00 UTC+2) in server time ---
    datetime server_midnight = iTime(Symbol(), PERIOD_D1, 0);
    if(server_midnight == 0) {
        Print("Error getting server midnight time for timings calculation.");
        g_last_initialized_day_time = 0; // Mark initialization failed for the day
        return;
    }

    // Target hour in GMT (13:00 UTC+2 is 11:00 GMT -> 13 - 2 = 11)
    int target_hour_gmt = utcPlus2_KeyHour_1300 - 2; // Assuming UTC+2 is always GMT+2

    // Calculate target hour in Server Time relative to server midnight
    int target_server_hour = target_hour_gmt + server_GMT_Offset_Manual;

    // Ensure target_server_hour is within 0-23
    target_server_hour = target_server_hour % 24;
    if(target_server_hour < 0) target_server_hour += 24;

    // Calculate the exact datetime in server time for the key time
    g_TodayKeyTime_Server = server_midnight + (long)(target_server_hour * 3600 + utcPlus2_KeyMinute_1300 * 60);


    //--- Calculate the observation end time (after observation duration) in server time ---
    g_ObservationEndTime_Server = g_TodayKeyTime_Server + observation_Duration_Minutes * 60;


     //--- Calculate the specific 15th candlestick entry time in server time ---
    // Find the bar index that starts exactly at or immediately after the Key Time (g_TodayKeyTime_Server)
    // We want the 15th M5 bar starting *from* the bar at g_TodayKeyTime_Server as the first bar.
    int key_bar_idx_for_count = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);

    if (key_bar_idx_for_count >= 0) // Ensure the starting bar for the count exists
    {
        // The index of the target bar relative to chart start is Key bar index + (Target count - 1)
        int target_entry_bar_idx = key_bar_idx_for_count + (entry_Candlestick_Index -1);

        // Get the time of the target entry bar (assuming enough bars exist)
        if(target_entry_bar_idx >= 0 && target_entry_bar_idx < Bars(Symbol(), Period()))
        {
             g_EntryTiming_Server = iTime(Symbol(), Period(), target_entry_bar_idx);
             Print(visual_comment_text + " - Calculated Target 15th M5 Candlestick Entry Time (Server Time): ", TimeToString(g_EntryTiming_Server, TIME_DATE|TIME_MINUTES));
        }
         else
         {
              Print("Warning: Not enough historical bars to determine the 15th candlestick time based on today's key time (idx: ", target_entry_bar_idx, ", bars: ", Bars(Symbol(),Period()), ").");
              g_EntryTiming_Server = 0; // Indicate timing not calculated
         }

         // --- Get the price at the Key Time bar for reference ---
         // Find the bar index at the exact Key Time (g_TodayKeyTime_Server) or the first one after
         int key_price_bar_index = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);
         if(key_price_bar_index >= 0 && key_price_bar_index < Bars(Symbol(), Period()))
         {
              g_KeyPrice_At_KeyTime = iOpen(Symbol(), Period(), key_price_bar_index); // Use Open price of the bar at the Key Time
              Print(visual_comment_text + " - Price at Key Time bar open: ", DoubleToString(g_KeyPrice_At_KeyTime, Digits()));
         }
         else
         {
              Print("Warning: Could not find a bar at or after the calculated Key Time to get the price.");
              g_KeyPrice_At_KeyTime = 0;
         }

    } else {
         Print("Warning: Could not find a bar at or after the calculated Key Time (", TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES), ") to start the 15-bar count. Insufficient data?");
         g_EntryTiming_Server = 0;
         g_KeyPrice_At_KeyTime = 0;
    }

    //--- Draw Visual Lines ---
    if (visual_enabled && visual_main_timing_lines)
    {
        // Ensure previous lines are removed before drawing new ones for the day
        RemoveDailyVisualObjects(); // This also removes previous OB visuals


        // Draw Vertical Line at Key Time (13:00 UTC+2 Server Time)
        // Using unique name based on current server time milliseconds + random for redraw safety
        string key_time_vline_name = "VLine_KeyTime_" + Symbol() + "_" + Period(); // Base name
        if (ObjectFind(0, key_time_vline_name) > 0) ObjectDelete(0, key_time_vline_name); // Delete existing
        if (g_TodayKeyTime_Server > 0 && ObjectCreate(0, key_time_vline_name, OBJ_VLINE, 0, g_TodayKeyTime_Server))
        {
            ObjectSetInteger(0, key_time_vline_name, OBJPROP_COLOR, vline_keytime_color);
            ObjectSetString(0, key_time_vline_name, OBJPROP_TEXT, visual_comment_text + ": 13:00 UTC+2 (" + TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES) + " Server)");
            ObjectSetInteger(0, key_time_vline_name, OBJPROP_RAY, false);
            ObjectSetInteger(0, key_time_vline_name, OBJPROP_SELECTABLE, false);
            ObjectSetInteger(0, key_time_vline_name, OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
        }


        // Draw Vertical Line at Observation End Time (Server Time)
         string obs_end_vline_name = "VLine_ObsEnd_" + Symbol() + "_" + Period(); // Base name
        if (ObjectFind(0, obs_end_vline_name) > 0) ObjectDelete(0, obs_end_vline_name); // Delete existing
        if (g_ObservationEndTime_Server > 0 && ObjectCreate(0, obs_end_vline_name, OBJ_VLINE, 0, g_ObservationEndTime_Server))
        {
             ObjectSetInteger(0, obs_end_vline_name, OBJPROP_COLOR, vline_obsend_color);
            ObjectSetString(0, obs_end_vline_name, OBJPROP_TEXT, visual_comment_text + ": Obs End (" + TimeToString(g_ObservationEndTime_Server, TIME_DATE|TIME_MINUTES) + " Server)");
            ObjectSetInteger(0, obs_end_vline_name, OBJPROP_RAY, false);
            ObjectSetInteger(0, obs_end_vline_name, OBJPROP_SELECTABLE, false);
             ObjectSetInteger(0, obs_end_vline_name, OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
        }
         ChartRedraw(0); // Redraw after drawing lines


        // Draw Optional Horizontal Line at Key Price
        if (visual_enabled && visual_key_price_line && g_KeyPrice_At_KeyTime > 0)
        {
             string key_price_hline_name = "HLine_KeyPrice_" + Symbol() + "_" + Period(); // Base name
            if (ObjectFind(0, key_price_hline_name) > 0) ObjectDelete(0, key_price_hline_name); // Delete existing
            if (ObjectCreate(0, key_price_hline_name, OBJ_HLINE, 0, g_KeyPrice_At_KeyTime))
            {
                ObjectSetInteger(0, key_price_hline_name, OBJPROP_COLOR, hline_keyprice_color);
                ObjectSetString(0, key_price_hline_name, OBJPROP_TEXT, visual_comment_text + ": Key Price");
                ObjectSetInteger(0, key_price_hline_name, OBJPROP_SELECTABLE, false);
                 ObjectSetInteger(0, key_price_hline_name, OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
                 ChartRedraw(0);
            }
        }
    }

     // --- Update the last initialized day marker ---
    g_last_initialized_day_time = server_midnight;

    // Reset static OB count arrays. They are defined globally, this resets the "active" count.
    g_bullishOB_count = 0;
    g_bearishOB_count = 0;
}


//+------------------------------------------------------------------+
//| Checks for a new trading day and performs reset/recalc         |
//| This function runs from OnTimer to ensure daily resets.         |
//+------------------------------------------------------------------+
void IsNewDayCheckAndReset()
{
    datetime server_midnight_today = iTime(Symbol(), PERIOD_D1, 0); // Server midnight of the current day

    // If the server midnight bar time is later than the last time we initialized/reset
    if(server_midnight_today > g_last_initialized_day_time && g_last_initialized_day_time != 0)
    {
        Print("--- New Day Detected. Resetting EA state for ", TimeToString(server_midnight_today, TIME_DATE), " ---");

        // --- Reset daily state variables ---
        g_InitialBias = 0;
        g_KeyPrice_At_KeyTime = 0;
        g_last_processed_bar_index = -1; // Ensure first bar of new day is processed
        g_bias_determined_today = false;
        g_entry_timed_window_alerted = false; // Allow the entry window check/alert on the new day
        g_order_blocks_scanned_today = false; // Allow OB detection for the new day

        // --- Recalculate and redraw visuals for the new day (cleans old visuals internally) ---
        CalculateAndDrawDailyTimings(); // This function also updates g_last_initialized_day_time and resets OB counts
    }
     else if (g_last_initialized_day_time == 0) // Special case for the very first time OnInit is called or after EA restart
     {
          g_last_initialized_day_time = server_midnight_today;
     }
}


//+------------------------------------------------------------------+
//| Function to remove all daily visual objects created by this EA   |
//| Removes vertical lines, horizontal lines, and Order Block visuals|
//+------------------------------------------------------------------+
void RemoveDailyVisualObjects()
{
     long chart_id = 0; // Current chart
     string name_prefix = visual_comment_text + ":"; // Standard prefix for our comments
     for (int i = ObjectsTotal(chart_id) - 1; i >= 0; i--)
     {
         string obj_name = ObjectName(chart_id, i);

          // Remove our specifically named lines (using substring search)
         if (StringFind(obj_name, "VLine_KeyTime_", 0) == 0 ||
             StringFind(obj_name, "VLine_ObsEnd_", 0) == 0 ||
             StringFind(obj_name, "HLine_KeyPrice_", 0) == 0)
         {
             ObjectDelete(chart_id, obj_name);
         }
         // Remove our OB rectangles and labels (using substring search based on objectName structure in st_OrderBlock)
          if (StringFind(obj_name, "BullOB_", 0) == 0 ||
             StringFind(obj_name, "BearOB_", 0) == 0 ||
             StringFind(obj_name, "_Label", 0) > 0 && (StringFind(obj_name, "BullOB_", 0) == 0 || StringFind(obj_name, "BearOB_", 0) == 0)) // Also covers labels
         {
             ObjectDelete(chart_id, obj_name);
         }

     }
     ChartRedraw(0); // Redraw after deletion
}


//+------------------------------------------------------------------+
//| Update mitigation status of detected Order Blocks (Stub/Basic)   |
//| Checks if price traded completely through the range of detected OBs|
//| Called on each new closed bar. Handles both Rectangles & Stored data|
//+------------------------------------------------------------------+
void UpdateMitigationStatus(int current_closed_bar_index)
{
     // Iterate through the stored static arrays for both types of OBs
     for (int i = 0; i < g_bullishOB_count; i++)
     {
         // Ensure the OB hasn't been mitigated already
         if (!g_bullishOrderBlocks[i].isMitigated)
         {
              // Check if the *previous closed bar's* Low went below the Bullish OB's Low
              // Check against index 1 (previous closed bar)
              if (iLow(Symbol(), Period(), 1) < g_bullishOrderBlocks[i].low - _Point * 2) // Use a small buffer below OB low
              {
                  g_bullishOrderBlocks[i].isMitigated = true;
                  Print(visual_comment_text + " - Bullish Order Block at ", TimeToString(g_bullishOrderBlocks[i].startTime, TIME_DATE|TIME_MINUTES), " mitigated.");

                  // Update visual object color if drawn
                   if(visual_enabled && visual_order_blocks && ObjectFind(0, g_bullishOrderBlocks[i].objectName) > 0)
                  {
                      ObjectSetInteger(0, g_bullishOrderBlocks[i].objectName, OBJPROP_COLOR, ob_mitigated_color);

                      if (ObjectFind(0, g_bullishOrderBlocks[i].labelName))
                      {
                            ObjectSetString(0, g_bullishOrderBlocks[i].labelName, OBJPROP_TEXT, "Bull OB Mitigated");
                            ObjectSetInteger(0, g_bullishOrderBlocks[i].labelName, OBJPROP_COLOR, clrGray);
                      }
                      ChartRedraw(0);
                  }
              }
         }
     }

     for (int i = 0; i < g_bearishOB_count; i++)
     {
          if (!g_bearishOrderBlocks[i].isMitigated)
          {
              // Check if the *previous closed bar's* High went above the Bearish OB's High
               if (iHigh(Symbol(), Period(), 1) > g_bearishOrderBlocks[i].high + _Point * 2) // Use a small buffer above OB high
              {
                   g_bearishOrderBlocks[i].isMitigated = true;
                   Print(visual_comment_text + " - Bearish Order Block at ", TimeToString(g_bearishOrderBlocks[i].startTime, TIME_DATE|TIME_MINUTES), " mitigated.");

                   if(visual_enabled && visual_order_blocks && ObjectFind(0, g_bearishOrderBlocks[i].objectName) > 0)
                  {
                      ObjectSetInteger(0, g_bearishOrderBlocks[i].objectName, OBJPROP_COLOR, ob_mitigated_color);
                        // Update label text and color
                      if (ObjectFind(0, g_bearishOrderBlocks[i].labelName))
                      {

                             ObjectSetInteger(0, g_bearishOrderBlocks[i].labelName, OBJPROP_COLOR, clrGray);
                      }
                       ChartRedraw(0);

//+------------------------------------------------------------------+
//| Basic Check and manage open positions (close at session end)     |
//| --- FOR FUTURE TRADE MANAGEMENT LOGIC ---                      |
//+------------------------------------------------------------------+
void ManageTrades()
{
     // Get current server time
     datetime now = TimeCurrent();

     // Calculate the session end hour in Server Time relative to server midnight
     // Convert Session End time from UTC+2 to GMT (22:00 UTC+2 is 20:00 GMT)
     int session_end_gmt_hour = session_End_UTC2_Hour - 2; // Assuming UTC+2 is GMT+2

     // Calculate the session end hour in Server Time relative to server midnight
     int session_end_server_hour = session_end_gmt_hour + server_GMT_Offset_Manual;
     // Ensure hour is within 0-23
     session_end_server_hour = session_end_server_hour % 24;
     if(session_end_server_hour < 0) session_end_server_hour += 24;

     // Calculate the exact session end datetime for TODAY in server time
     datetime server_midnight_today = iTime(Symbol(), PERIOD_D1, 0); // Server midnight of the current day
     datetime session_end_server_time = server_midnight_today + (long)(session_end_server_hour * 3600 + session_End_UTC2_Minute * 60);

     // Check if current time has passed the session end time for today
     if(now >= session_end_server_time)
     {
         // Check all open positions for this symbol and magic number
         for (int i = PositionsTotal() - 1; i >= 0; i--)
         {
             ulong position_ticket = PositionGetTicket(i);
             if (PositionGetString(POSITION_SYMBOL) == Symbol() && PositionGetInteger(POSITION_MAGIC) == magic_Number)
