void delay()
  {
   datetime _opened_last_time = TimeCurrent() ;
   for(int i=0; i<OrdersHistoryTotal(); i++)
     {
      if(OrderSelect(i,SELECT_BY_POS,MODE_HISTORY)==false)
         continue;
      if(OrderCloseTime() < _opened_last_time)
         continue;

      DeleteAllOrders();
      Print(" going to sleep now");
      Sleep(Time_To_Wait*1000);
      Print("Wake up!");
     }
  }

void Update_SS_And_BS_Objects()
  {
   double  Ask=SymbolInfoDouble(Symbol(), SYMBOL_ASK);
   double  Bid=SymbolInfoDouble(Symbol(), SYMBOL_BID);
   double stops=SymbolInfoInteger(Symbol(),SYMBOL_TRADE_STOPS_LEVEL) * Point();

   double bs = ObjectGetDouble(0,HL_BS, OBJPROP_PRICE);           //Old buyStop Value
   double OrderDistance = NormalizeDouble(Distance * Point(), Digits()); //how far against the current price to place order
   double NewValue = NormalizeDouble(Ask+OrderDistance,Digits());   //New buystop Value
   double OldValue=NormalizeDouble(bs,Digits());                    //Normalized Old Value
   if(NewValue<OldValue)
     {
      ObjectDelete(0,HL_BS);

      ObjectCreate(0,HL_BS,OBJ_HLINE,0,TimeCurrent(),NewValue);
      ObjectSetInteger(0, HL_BS, OBJPROP_STYLE, STYLE_SOLID);
      ObjectSetInteger(0, HL_BS, OBJPROP_COLOR, Lime);
      ObjectSetInteger(0, HL_BS, OBJPROP_BACK, true);
      ObjectSetInteger(0, HL_BS, OBJPROP_WIDTH, 2);
     }

   double ss = ObjectGetDouble(0,HL_SS, OBJPROP_PRICE);  //Old sellStop Value
   OldValue=NormalizeDouble(ss,Digits());                  //Normalized old Vlaue
   NewValue = NormalizeDouble(Bid-OrderDistance,Digits()); //New Value
   if(ss>0 && NewValue>OldValue)
     {
      ObjectDelete(0,HL_SS);

      ObjectCreate(0,HL_SS,OBJ_HLINE,0,TimeCurrent(),NewValue);
      ObjectSetInteger(0, HL_SS, OBJPROP_STYLE, STYLE_SOLID);
      ObjectSetInteger(0, HL_SS, OBJPROP_COLOR, Crimson);
      ObjectSetInteger(0, HL_SS, OBJPROP_BACK, true);
      ObjectSetInteger(0, HL_SS, OBJPROP_WIDTH, 2);
     }
  }

int StoreAndPlaceDelayedOrders(int order)
  {
   double  Ask=SymbolInfoDouble(Symbol(), SYMBOL_ASK);
   double  Bid=SymbolInfoDouble(Symbol(), SYMBOL_BID);
   double stops=SymbolInfoInteger(Symbol(),SYMBOL_TRADE_STOPS_LEVEL) * Point();

   int copiedL=CopyLow(_Symbol,PERIOD_CURRENT,0,3,Low);
   double Open[];
   int copiedO=CopyOpen(_Symbol,PERIOD_CURRENT,0,3,Open);

//if there are no signals already delayed store the signal so we can open it again in the future
   if(DelayedSignalTime == 0 && order >= 0)     {
      //store the current signal
      DelayedSignalType = order;
      DelayedSignalTime = iTime(NULL,0,0);

      //print message notifying user about the delay
      if(DelayedSignalType == 0)
        {
         Print("A buy order @Time: ",iTime(NULL,1,0)," has been delayed till next candle");
         Comment("A buy order @Time: ",iTime(NULL,1,0)," has been delayed till next candle");
        }
      else        {
         Print("A sell order @Time: ",iTime(NULL,1,0)," has been delayed till next candle");
         Comment("A sell order @Time: ",iTime(NULL,1,0)," has been delayed till next candle");
        }

      return -1; //return -1 because order has been delayed to prevent order from being placed
     }

//make sure that the delayed signal is rady to be placed on a different bar, then place it
   if(DelayedSignalTime != iTime(NULL,0,0) && DelayedSignalTime !=0)
     {
      /*
        PULLBACK logic such that,when a pullback occurs in number of pips and market price returned back
        on same candle were pullback counting began, and a new candle was form,let TRADING signal be
        cancelled .this is when WAIT/DELAY TILL NEXT CANDLE =TRUE
      */


      if(!Indicator_Arrow_Repaint && ((DelayedSignalType == 1 && High[0] >= High[1]) || (DelayedSignalType == 0 && Low[0] <= Low[1])))
        {
         //delete all objects and reset variables that determine the delayed signal and virtual pending stops
         ObjectDelete(0, HL_SS);
         ObjectDelete(0, HL_BS);
         DelayedSignalTime = 0;

         Print("Delayed Order Cancelled Since market price returned back on same candle were pullback counting began");
         Comment("Delayed Order Cancelled Since market price returned back on same candle were pullback counting began");

         return -1; //return no signal
        }

      /*
        another pullback condition to work after WAIT/DELAY TILL NEXT CANDLE =TRUE,
        the new candle pullback setting should work with either ARROW REPAINT OR PULLBACK.
        when WAIT/DELAY TILL NEXT CANDLE =TRUE,Add new setting , NUMBER OF PIPS PULLBACK IN NEXT CANDLE.
        this setting in pips will allow ea to calculate number of pips pullback in new candle before allowing
        pending stop be opened.New condition is, untill a pullback in number of pips defined in next candle
        before ea opens virtual pending stop.but when this is set to =0,its deactivated
      */
      if(PullBackInNextCandle > 0 && ((DelayedSignalType == 1 &&  Open[0]-Low[0] < PullBackInNextCandle*10*Point()) || (DelayedSignalType == 0 && High[0]-Open[0] < PullBackInNextCandle*10*Point())))
         return -1;                                                          //return no signal since pull back hasnt been satisfied
      //---------------------------
      if(PullBackInNextCandle > 0 && DelayedSignalTime < iTime(NULL,0,1))   //cancel the order since pull back wasnt satisfied and a new second bar was formed so signal is no longer valid
        {
         //delete all objects and reset variables that determine the delayed signal and virtual pending stops
         ObjectDelete(0, HL_SS);
         ObjectDelete(0, HL_BS);
         DelayedSignalTime = 0;

         Print("Delayed Order Cancelled Since Pullback in was not enough on the next candle till it closed");
         Comment("Delayed Order Cancelled Since Pullback in was not enough on the next candle till it closed");

         return -1; //return no signal
        }

      //if we reach this point place back the delayed signal since all the delay pull back conditions were met
      DelayedSignalTime = 0; // reset time so that we do not place the order again

      //notify user about the order, that it will be resumed
      if(order == 0)        {
         Print("A delayed buy order from Time: ",DelayedSignalTime," will now be placed");
         Comment("A delayed buy order from Time: ",DelayedSignalTime," will now be placed");
        }
      else        {
         Print("A delayed sell order from Time: ",DelayedSignalTime," will now be placed");
         Comment("A delayed sell order from Time: ",DelayedSignalTime," will now be placed");
        }
      return DelayedSignalType; //retrieve the placed order so that it can be placed
     }

   return -1; //return -1 because order has been delayed already or there was no signal
  }

//|          Determine signal with arrow repaint stratey              |
//|  we get the signals from the indicator, then determine how many  |
//|  times theindicator repaints the same signal and it repaints a   |
//|           certain times we place a buy or sell order             |
//+------------------------------------------------------------------+
int ArrowRepaintStratey(double sellValue, double buyValue)
  {
//if there are no signals from the indicator we make sure that we reset the repaint values
   if(buyValue == 0 && sellValue == 0)
     {
      current_Repaint = 0;
      current_signal_value = 0;
     }

   else //if there a sinals from indicator
     {
      //we reset repaint values if last signal was buy and current signal is sell and vice versa for last sell and current by
      //this makes sure we only count repaints for the current sinal only
      if((buyValue > 0 && !Last_Signal_buy) || (sellValue > 0 && Last_Signal_buy))
        {
         current_Repaint = 0;
         current_signal_value = 0;
        }

      if(buyValue > 0) //current signal is buy
        {
         Last_Signal_buy = true; //we mke sure that we indicate the signal is a buy for the next call

         //set the current signal value to indicator buy value we dont count the first signal occurence as repaint
         if(current_signal_value == 0)
            current_signal_value = buyValue;

         //if current signal value is different then there was a repaint and we adjust values accordingly
         if(current_signal_value != buyValue)
           {
            current_Repaint++;
            current_signal_value = buyValue;
           }

         Comment("Buy Repaint: ",current_Repaint);
         Print("Buy Repaint: ",current_Repaint);

         //repaint requirements are met so we reset values and return a buy back so that e can place order
         if(current_Repaint >= Number_Of_Repaint)
           {
            current_Repaint = 0;
            current_signal_value = 0;
            return OP_BUY;
           }
        }

      if(sellValue > 0) //current signal is sell
        {
         Last_Signal_buy = false; //we mke sure that we indicate the signal is a sell for the next call

         //set the current signal value to indicator buy value we dont count the first signal occurence as repaint
         if(current_signal_value == 0)
            current_signal_value = sellValue;

         //if current signal value is different then there was a repaint and we adjust values accordingly
         if(current_signal_value != sellValue)
           {
            current_Repaint++;
            current_signal_value = sellValue;
           }


         Comment("Sell Repaint: ",current_Repaint);
         Print("Sell Repaint: ",current_Repaint);

         //repaint requirements are met so we reset values and return a buy back so that e can place order
         if(current_Repaint >= Number_Of_Repaint)
           {
            current_Repaint = 0;
            current_signal_value = 0;
            return OP_SELL;
           }
        }
     }

   return -1; //if e get to this point then there as never a signal or repaint taret hasnt been met yet
  }

  {

   datetime Time[];

   int copiedC=CopyClose(_Symbol,PERIOD_CURRENT,0,3,Close);

   double Low[];
   int copiedL=CopyLow(_Symbol,PERIOD_CURRENT,0,3,Low);
   double Open[];
   int copiedO=CopyOpen(_Symbol,PERIOD_CURRENT,0,3,Open);


   double barSize      = iHigh(NULL,0,0) - iLow(NULL,0,0);
   double filterPoints = NormalizeDouble(Point() * Filter, Digits());

   static int      Delay2 = 0;
   static datetime delay_ = 0;

   bool Sell = false;
   bool Buy  = false;

   if(High[0]==sellValue && (High[0]- Close[0])>=(PullBack*Point()))
     {
      Sell = true;
      if(delay_==0)
         delay_ = TimeCurrent();
     }

   if(Low[0]==buyValue && (Close[0]- Low[0])>=(PullBack*Point()))
     {
      Buy = true;
      if(delay_==0)
         delay_ = TimeCurrent();
     }

   if(Sell==false && Buy==false)
     {
      Delay2 = 0;
      delay_ = 0;
     }
   else
      Delay2 = (int)(TimeCurrent() - delay_);

   if(barSize > Filter || Filter==0)
     {
      if(Buy==true && Delay2>=delay_seconds)
         return OP_BUY;
      else
         if(Sell==true && Delay2>=delay_seconds)
            return OP_SELL;
     }

   return -1;
  }

 // Helper function to convert string to wide char array (ushort)
void StringToWideCharArray(const string &src, ushort &dest[])
{
    int length = StringLen(src);
    ArrayResize(dest, length + 1);  // Resize the destination array to fit the string + null-terminator

    for(int i = 0; i < length; i++)
    {
        dest[i] = (ushort)src[i];  // Convert each character to a wide character
    }

    dest[length] = 0;  // Null-terminate the wide character array
}
