Okay, here is version 1 of the MQL5 Expert Advisor for the core "714 Method" implementation.

This version focuses on the time zone synchronization, drawing visual markers on the chart, determining the initial daily bias based on the post-key time movement, and identifying the specific "15th candlestick" entry window. It *does not* include the sophisticated OB/FVG detection or automatic trade placement/management (beyond closing at the end of the day), leaving those as placeholders for your integration in future versions.

Remember to accurately determine your broker's server time offset from GMT (UTC+0) and input it in the `server_GMT_Offset_Manual` parameter.

    {
        Print("Warning: Failed to initialize CTrade object. Trading operations will not work.");
        // Continue, as visuals and analysis can still function
    }

void OnTimer()

    // --- Check for New Day and Re-initialize Timings if needed ---
    IsNewDayCheckAndReset();

    // --- Process logic only on a NEW M5 closed bar ---
    // Bars(Symbol(), Period()) gives total number of bars. Index of the last closed bar is total_bars - 1.
    int total_bars = Bars(Symbol(), Period());
    if (total_bars <= 1) return; // Need at least two bars to have a closed bar at index 0.

    int closed_bar_index = total_bars - 1;

    if (closed_bar_index <= g_last_processed_bar_index)
    {
         // No new bar has closed since the last check, or historical bars loading
         return;
    }

    // This is a new closed bar that we haven't processed for our main logic
    g_last_processed_bar_index = closed_bar_index;
    datetime current_closed_bar_time = iTime(Symbol(), Period(), closed_bar_index);

    // Print("Processing New Bar at Server Time: ", TimeToString(current_closed_bar_time, TIME_DATE|TIME_MINUTES));


    //--- Step 4: Initial Trend Observation (Determine Bias) ---
    // Run this logic only once per day, triggered by the first closed bar
    // that is at or after the Observation End Time.
    if (!g_bias_determined_today && current_closed_bar_time >= g_ObservationEndTime_Server)
    {
         // Find the bar index that starts exactly at or immediately after the Key Time (g_TodayKeyTime_Server)
         int key_bar_idx = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);

         // Find the bar index that starts exactly at or immediately after the Observation End Time (g_ObservationEndTime_Server)
         int obs_end_bar_idx = iBarShift(Symbol(), Period(), g_ObservationEndTime_Server, false);

         // Ensure we found both relevant bars and the end bar index is valid and after the key bar index
         if (key_bar_idx >= 0 && obs_end_bar_idx >= 0 && obs_end_bar_idx < total_bars && obs_end_bar_idx >= key_bar_idx)
         {
             // We compare the Close of the bar at the Observation End to the Open of the bar at the Key Time
             double price_at_obs_end = iClose(Symbol(), Period(), obs_end_bar_idx);
             double price_at_keytime = iOpen(Symbol(), Period(), key_bar_idx);

             if (price_at_obs_end > price_at_keytime)
             {
                 g_InitialBias = 1; // Market moved up after the Key Time
                 Print(visual_comment_text + " - Initial Bias determined: BULLISH movement after Key Time. Preparing to look for SELL opportunities.");
             }
             else if (price_at_obs_end < price_at_keytime)
             {
                 g_InitialBias = -1; // Market moved down after the Key Time
                 Print(visual_comment_text + " - Initial Bias determined: BEARISH movement after Key Time. Preparing to look for BUY opportunities.");
             }
             else

             }

//+------------------------------------------------------------------+
//| Function to perform daily calculation and visualize timings      |
//+------------------------------------------------------------------+
void CalculateAndDrawDailyTimings()
{
    // --- Calculate today's Primary Key Time (13:00 UTC+2) in server time ---
    // Get the start of the current *server* day
    datetime server_midnight = iTime(Symbol(), PERIOD_D1, 0);
    if(server_midnight == 0) {
        Print("Error getting server midnight time.");
        return; // Cannot calculate without this
    }

    // Target hour in GMT (13:00 UTC+2 is 11:00 GMT -> 13 - 2 = 11)
    int target_hour_gmt = utcPlus2_KeyHour - 2; // Assuming UTC+2 is always GMT+2

    // Calculate target hour in Server Time relative to server midnight
    int target_server_hour = target_hour_gmt + server_GMT_Offset_Manual;

    // Ensure target_server_hour is within 0-23
    target_server_hour = target_server_hour % 24;
    if(target_server_hour < 0) target_server_hour += 24;

    // Calculate the exact datetime in server time for the key time
    g_TodayKeyTime_Server = server_midnight + (long)(target_server_hour * 3600 + utcPlus2_KeyMinute * 60);


    //--- Calculate the observation end time (after observation duration) in server time ---
    g_ObservationEndTime_Server = g_TodayKeyTime_Server + observation_Duration_Minutes * 60;

    //--- Calculate the specific 15th candlestick entry time in server time ---
    // Find the bar index that starts exactly at or immediately after the Key Time (g_TodayKeyTime_Server)
    // We want the 15th M5 bar starting *from* the bar at g_TodayKeyTime_Server as the first bar.
    int key_bar_idx_for_count = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);

    if (key_bar_idx_for_count >= 0) // Ensure the starting bar for the count exists
    {
        // The index of the 15th bar after key_bar_idx_for_count is key_bar_idx_for_count + (15 - 1) = key_bar_idx_for_count + 14
        int target_entry_bar_idx = key_bar_idx_for_count + (entry_Candlestick_Index -1); // Calculate the index

        // Get the time of the target entry bar (assuming enough bars exist)
        if(target_entry_bar_idx >= 0 && target_entry_bar_idx < Bars(Symbol(), Period()))
        {
             g_EntryTiming_Server = iTime(Symbol(), Period(), target_entry_bar_idx);
             Print(visual_comment_text + " - Target 15th M5 Candlestick Entry Time (Server Time): ", TimeToString(g_EntryTiming_Server, TIME_DATE|TIME_MINUTES));
        }
         else
         {
              Print("Warning: Not enough historical bars to determine the 15th candlestick time based on today's key time.");
              g_EntryTiming_Server = 0; // Indicate timing not calculated
         }

         // --- Get the price at the Key Time bar for reference ---
         // Find the bar index at the exact Key Time (g_TodayKeyTime_Server) or the first one after
         int key_price_bar_index = iBarShift(Symbol(), Period(), g_TodayKeyTime_Server, false);
         if(key_price_bar_index >= 0 && key_price_bar_index < Bars(Symbol(), Period()))
         {
              g_KeyPrice_At_KeyTime = iOpen(Symbol(), Period(), key_price_bar_index); // Use Open price of the bar at the Key Time
              Print(visual_comment_text + " - Price at Key Time bar open: ", DoubleToString(g_KeyPrice_At_KeyTime, Digits()));
         }
         else
         {
              Print("Warning: Could not find a bar at or after the calculated Key Time to get the price.");
              g_KeyPrice_At_KeyTime = 0;
         }

    } else {
         Print("Warning: Could not find a bar at or after the calculated Key Time (", TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES), ") to start the 15-bar count.");
         g_EntryTiming_Server = 0;
         g_KeyPrice_At_KeyTime = 0;
    }

    //--- Draw Visual Lines ---
    if (visual_lines_enabled)
    {
        // Ensure previous lines are removed before drawing new ones for the day
        RemoveDailyVisualObjects();

        // Draw Vertical Line at Key Time (13:00 UTC+2 Server Time)
        if (g_TodayKeyTime_Server > 0 && ObjectCreate(0, visuals_keytime_name, OBJ_VLINE, 0, g_TodayKeyTime_Server))
        {
            ObjectSetInteger(0, visuals_keytime_name, OBJPROP_COLOR, vline_keytime_color);
            ObjectSetString(0, visuals_keytime_name, OBJPROP_TEXT, visual_comment_text + ": 13:00 UTC+2 (" + TimeToString(g_TodayKeyTime_Server, TIME_DATE|TIME_MINUTES) + " Server)");
            ObjectSetInteger(0, visuals_keytime_name, OBJPROP_RAY, false);
            ObjectSetInteger(0, visuals_keytime_name, OBJPROP_SELECTABLE, false);
            ObjectSetInteger(0, visuals_keytime_name, OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
        }

        // Draw Vertical Line at Observation End Time (Server Time)
        if (g_ObservationEndTime_Server > 0 && ObjectCreate(0, visuals_obsend_name, OBJ_VLINE, 0, g_ObservationEndTime_Server))
        {
            ObjectSetInteger(0, visuals_obsend_name, OBJPROP_COLOR, vline_obsend_color);
            ObjectSetString(0, visuals_obsend_name, OBJPROP_TEXT, visual_comment_text + ": Obs End (" + TimeToString(g_ObservationEndTime_Server, TIME_DATE|TIME_MINUTES) + " Server)");
            ObjectSetInteger(0, visuals_obsend_name, OBJPROP_RAY, false);
            ObjectSetInteger(0, visuals_obsend_name, OBJPROP_SELECTABLE, false);
             ObjectSetInteger(0, visuals_obsend_name, OBJPROP_HIDDEN, true); // Hide on lower/higher TFs
        }
         
        // Draw Optional Horizontal Line at Key Price
        if (visual_key_price_line && g_KeyPrice_At_KeyTime > 0 && ObjectCreate(0, visuals_keyprice_name, OBJ_HLINE, 0, g_KeyPrice_At_KeyTime))
        {
            ObjectSetInteger(0, visuals_keyprice_name, OBJPROP_COLOR, hline_keyprice_color);
            ObjectSetString(0, visuals_keyprice_name, OBJPROP_TEXT, visual_comment_text + ": Key Price");
            ObjectSetInteger(0, visuals_keyprice_name, OBJPROP_SELECTABLE, false);
        }

         // Optionally add more visual lines for other key times (7AM, 4PM UTC+2 equivalents) if needed

    }

     // --- Update the last initialized day marker ---
    g_last_initialized_day_time = server_midnight;
}


//+------------------------------------------------------------------+
//| Checks for a new trading day and performs reset/recalc         |
//+------------------------------------------------------------------+
void IsNewDayCheckAndReset()
{
    datetime server_midnight_today = iTime(Symbol(), PERIOD_D1, 0); // Server midnight of the current day

    // If the server midnight bar time is later than the last time we initialized/reset
    if(server_midnight_today > g_last_initialized_day_time)
    {
        Print("--- New Day Detected. Resetting EA state for ", TimeToString(server_midnight_today, TIME_DATE), " ---");

        // --- Reset daily state variables ---
        g_InitialBias = 0;
        g_KeyPrice_At_KeyTime = 0;
        g_last_processed_bar_index = -1; // Ensure first bar of new day is processed
        g_bias_determined_today = false;
        g_entry_timed_window_alerted = false; // Allow the entry window alert on the new day

        // --- Recalculate and redraw visuals for the new day ---
        CalculateAndDrawDailyTimings();

        // --- Update the last initialized day marker ---
        g_last_initialized_day_time = server_midnight_today;
    }
}
