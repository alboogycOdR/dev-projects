// Session detection
is_asian_session() =>
    hour >= 21 or hour < 8

is_london_session() =>
    hour >= 2 and hour < 11

is_newyork_session() =>
    hour >= 8 and hour < 17

get_current_killzone() =>
    if is_london_session()
        "London"
    else if is_newyork_session()
        "New York"
    else
        "None"

// Swing point detection
is_swing_high(len=5) =>
    high[len] > high[len+1] and high[len] > high[len+2] and 
    high[len] > high[len-1] and high[len] > high[len-2]

is_swing_low(len=5) =>
    low[len] < low[len+1] and low[len] < low[len+2] and 
    low[len] < low[len-1] and low[len] < low[len-2]

htf_low = request.security(syminfo.tickerid, htf_tf, low, lookahead=barmerge.lookahead_off)
htf_close = request.security(syminfo.tickerid, htf_tf, close, lookahead=barmerge.lookahead_off)

// Correlated Asset Data
corr_high = request.security(correlated_asset, timeframe.period, high, lookahead=barmerge.lookahead_off)
corr_low = request.security(correlated_asset, timeframe.period, low, lookahead=barmerge.lookahead_off)

// HTF Bias Determination
htf_bias := htf_close > htf_close[1] ? "BULLISH" : htf_close < htf_close[1] ? "BEARISH" : htf_bias[1]

if is_asian_session()

        london_high := high
    if na(london_low) or low < london_low
        london_low := low

if dayofweek != dayofweek[1]

m5_low = request.security(syminfo.tickerid, "5", low[1], lookahead=barmerge.lookahead_off)
m5_high_2 = request.security(syminfo.tickerid, "5", high[2], lookahead=barmerge.lookahead_off)
m5_low_2 = request.security(syminfo.tickerid, "5", low[2], lookahead=barmerge.lookahead_off)
m5_close = request.security(syminfo.tickerid, "5", close, lookahead=barmerge.lookahead_off)

// Bullish FVG: low[0] > high[2]
bullish_fvg = m5_low > m5_high_2
// Bearish FVG: high[0] < low[2]
bearish_fvg = m5_high < m5_low_2

// Break of Structure Detection
detect_bos() =>
    swing_high_broken = close > last_swing_high and not na(last_swing_high)
    swing_low_broken = close < last_swing_low and not na(last_swing_low)
    
    bos_bull = swing_high_broken and htf_bias == "BULLISH"
    bos_bear = swing_low_broken and htf_bias == "BEARISH"
    
    [bos_bull, bos_bear]

// Market Structure Shift Detection
detect_mss() =>
    [bos_bull, bos_bear] = detect_bos()
    mss_bull = bos_bull and htf_bias == "BEARISH"  // BOS against trend
    mss_bear = bos_bear and htf_bias == "BULLISH"
    [mss_bull, mss_bear]

// Liquidity Sweep Detection
detect_liquidity_sweep() =>
    swept_asian_high = not na(asian_high) and high > asian_high and low < asian_high
    swept_asian_low = not na(asian_low) and low < asian_low and high > asian_low
    swept_london_high = not na(london_high) and high > london_high and low < london_high
    swept_london_low = not na(london_low) and low < london_low and high > london_low
    swept_pdh = not na(pdh) and high > pdh and low < pdh
    swept_pdl = not na(pdl) and low < pdl and high > pdl
    
    any_sweep = swept_asian_high or swept_asian_low or swept_london_high or swept_london_low or swept_pdh or swept_pdl
    any_sweep

// SMT Divergence Detection (Simplified)
detect_smt_divergence() =>
    // Primary makes new high, correlated fails to confirm
    primary_new_high = high > highest(high, 10)[1]
    corr_fails_high = corr_high < highest(corr_high, 10)[1]
    
    // Primary makes new low, correlated fails to confirm
    primary_new_low = low < lowest(low, 10)[1]
    corr_fails_low = corr_low > lowest(corr_low, 10)[1]
    
    smt_div = (primary_new_high and corr_fails_high) or (primary_new_low and corr_fails_low)
    smt_div

// Order Block Detection (Simplified)
detect_order_block() =>
    // Last opposing candle before strong move
    strong_bullish_move = close > open and (close - open) > (high - low) * 0.7 and close > close[1]
    strong_bearish_move = close < open and (open - close) > (high - low) * 0.7 and close < close[1]
    
    bullish_ob = strong_bullish_move[1] and close[1] < open[1]  // Previous candle was bearish
    bearish_ob = strong_bearish_move[1] and close[1] > open[1]  // Previous candle was bullish
    
    [bullish_ob, bearish_ob]

// IFVG Detection (Simplified)
detect_ifvg() =>
    // FVG that gets traded through then holds as support/resistance
    ltf_timeframe = ltf_tf == "1" ? "1" : "30S"
    
    ltf_high = request.security(syminfo.tickerid, ltf_timeframe, high, lookahead=barmerge.lookahead_off)
    ltf_low = request.security(syminfo.tickerid, ltf_timeframe, low, lookahead=barmerge.lookahead_off)
    ltf_close = request.security(syminfo.tickerid, ltf_timeframe, close, lookahead=barmerge.lookahead_off)
    
    // Simplified IFVG logic - price rejection from a previously violated level
    bullish_ifvg = ltf_low > ltf_low[1] and ltf_close > ltf_low[2]
    bearish_ifvg = ltf_high < ltf_high[1] and ltf_close < ltf_high[2]
    
    [bullish_ifvg, bearish_ifvg]

// Core SMC Model Logic
execute_core_smc_model() =>
    var bool local_event1 = false
    var bool local_event2 = false
    var bool local_event3 = false
    var bool local_event4 = false
    
    // Event 1: Manipulation Confirmed
    if not local_event1
        liquidity_sweep = detect_liquidity_sweep()
        smt_div = detect_smt_divergence()
        if liquidity_sweep or smt_div
            local_event1 := true
    
    // Event 2: Structure Break Confirmed
    if local_event1 and not local_event2
        [mss_bull, mss_bear] = detect_mss()
        [bos_bull, bos_bear] = detect_bos()
        if mss_bull or mss_bear or bos_bull or bos_bear
            local_event2 := true
    
    // Event 3: Return to HTF POI
    if local_event2 and not local_event3
        [bull_ob, bear_ob] = detect_order_block()
        if bullish_fvg or bearish_fvg or bull_ob or bear_ob
            local_event3 := true
    
    // Event 4: LTF Entry Confirmation
    if local_event3 and not local_event4
        [bull_ifvg, bear_ifvg] = detect_ifvg()
        if bull_ifvg or bear_ifvg
            local_event4 := true
    
    // Return completion status
    all_events_complete = local_event1 and local_event2 and local_event3 and local_event4
    
    // Reset if new bar starts a fresh setup
    if barstate.isconfirmed and all_events_complete
        local_event1 := false
        local_event2 := false
        local_event3 := false
        local_event4 := false
    
    [local_event1, local_event2, local_event3, local_event4, all_events_complete]

// ICT Silver Bullet Logic
execute_silver_bullet() =>
    var bool sb_event1 = false
    var bool sb_event2 = false
    
    // Must be in specific time window (10:00-11:00 EST example)
    in_sb_window = (hour >= 10 and hour < 11)
    
    if in_sb_window and not sb_event1
        liquidity_sweep = detect_liquidity_sweep()
        if liquidity_sweep
            sb_event1 := true
    
    if sb_event1 and not sb_event2
        if bullish_fvg or bearish_fvg
            sb_event2 := true
    
    sb_complete = sb_event1 and sb_event2
    
    // Reset for next opportunity
    if not in_sb_window
        sb_event1 := false
        sb_event2 := false
    
    [sb_event1, sb_event2, sb_complete]
