## **1. Bot Code (Python)**

```python
"""
Binance Perpetual Futures Liquidity Sweep + Market Structure Shift Bot
Author: Alister's ChatGPT Assistant
Date: 2025-08-09

Description:
    This bot implements a price-action-based strategy without indicators.
    It trades liquidity sweeps followed by market structure shifts
    on Binance USDT-Margined Perpetual Futures.

Requirements:
    pip install python-binance pandas numpy

⚠ DISCLAIMER:
    This bot is for educational purposes only.
    Live trading involves substantial risk — test thoroughly before real use.
"""

from binance.client import Client
from binance.enums import *
import pandas as pd
import numpy as np
import time
import datetime

# --------------------------------------
# USER CONFIGURATION
# --------------------------------------
API_KEY = "YOUR_API_KEY"
API_SECRET = "YOUR_API_SECRET"

SYMBOL = "BTCUSDT"          # Trading pair
INTERVAL = "1m"             # Candle interval for structure shift detection
LOOKBACK = 50               # Number of candles to analyze
RISK_PERCENT = 1            # Risk % per trade
LEVERAGE = 10               # Futures leverage

# --------------------------------------
# INITIALIZE CLIENT
# --------------------------------------
client = Client(API_KEY, API_SECRET)
client.futures_change_leverage(symbol=SYMBOL, leverage=LEVERAGE)

# --------------------------------------
# FUNCTIONS
# --------------------------------------
def get_klines(symbol, interval, limit):
    """Fetch recent kline/candlestick data from Binance."""
    data = client.futures_klines(symbol=symbol, interval=interval, limit=limit)
    df = pd.DataFrame(data, columns=[
        'timestamp', 'open', 'high', 'low', 'close', 'volume', 
        'close_time', 'quote_asset_volume', 'num_trades', 
        'taker_buy_base', 'taker_buy_quote', 'ignore'
    ])
    df['open'] = df['open'].astype(float)
    df['high'] = df['high'].astype(float)
    df['low'] = df['low'].astype(float)
    df['close'] = df['close'].astype(float)
    return df

def detect_liquidity_sweep(df):
    """
    Detect liquidity sweep:
    - Price breaks previous high/low
    - Quickly rejects back inside range
    """
    prev_high = df['high'].iloc[-3]
    prev_low = df['low'].iloc[-3]
    last_high = df['high'].iloc[-1]
    last_low = df['low'].iloc[-1]
    last_close = df['close'].iloc[-1]
    
    if last_high > prev_high and last_close < prev_high:
        return "short"  # Sweep of highs
    elif last_low < prev_low and last_close > prev_low:
        return "long"   # Sweep of lows
    else:
        return None

def calculate_position_size(balance, stop_distance):
    """Position size = (account risk in USDT) / stop distance."""
    risk_amount = balance * (RISK_PERCENT / 100)
    qty = risk_amount / stop_distance
    return round(qty, 3)

def place_order(direction, stop_loss, take_profit):
    """Place a futures market order with SL and TP."""
    balance = float(client.futures_account_balance()[0]['balance'])
    stop_distance = abs(stop_loss - float(client.futures_mark_price(symbol=SYMBOL)['markPrice']))
    qty = calculate_position_size(balance, stop_distance)

    side = SIDE_BUY if direction == "long" else SIDE_SELL
    opposite_side = SIDE_SELL if direction == "long" else SIDE_BUY

    # Entry
    client.futures_create_order(
        symbol=SYMBOL,
        side=side,
        type=ORDER_TYPE_MARKET,
        quantity=qty
    )

    # Stop Loss
    client.futures_create_order(
        symbol=SYMBOL,
        side=opposite_side,
        type=ORDER_TYPE_STOP_MARKET,
        stopPrice=stop_loss,
        closePosition=True
    )

    # Take Profit
    client.futures_create_order(
        symbol=SYMBOL,
        side=opposite_side,
        type=ORDER_TYPE_LIMIT,
        price=take_profit,
        quantity=qty,
        timeInForce=TIME_IN_FORCE_GTC
    )

    print(f"Order placed: {direction.upper()} | Qty: {qty} | SL: {stop_loss} | TP: {take_profit}")

# --------------------------------------
# MAIN LOOP
# --------------------------------------
print("Starting Binance Futures Liquidity Sweep Bot...")

while True:
    try:
        df = get_klines(SYMBOL, INTERVAL, LOOKBACK)
        signal = detect_liquidity_sweep(df)
        
        if signal:
            print(f"Signal detected: {signal.upper()} at {datetime.datetime.now()}")
            
            if signal == "long":
                stop_loss = df['low'].iloc[-1] - 10
                take_profit = df['close'].iloc[-1] + (stop_loss - df['close'].iloc[-1]) * -2
                place_order("long", stop_loss, take_profit)
            
            elif signal == "short":
                stop_loss = df['high'].iloc[-1] + 10
                take_profit = df['close'].iloc[-1] - (df['close'].iloc[-1] - stop_loss) * -2
                place_order("short", stop_loss, take_profit)

        time.sleep(5)

    except Exception as e:
        print(f"Error: {e}")
        time.sleep(5)
```

### **How It Works**

1. **Key levels**: The bot checks recent highs/lows for liquidity sweeps.
2. **Sweep detection**: If price sweeps a level and closes back inside, it’s a potential reversal.
3. **Entry**:

   * Long = sweep of lows + rejection
   * Short = sweep of highs + rejection
4. **Stop loss**: Placed just beyond the sweep.
5. **Take profit**: Risk\:Reward ≈ 1:2.
6. **Risk management**: Position size based on % account risk.
